solution: 

int get_score(int n) {
    int score = 0;
    while (n >= 10) {
        score += n % 10;
        n /= 10;
    }
    score += n;
    return score;
}
int main() {
    int n;
    cin >> n;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    int factor = 1;
    int best_factor = 1;
    int best_score = 1;
    int half = floor(sqrt(n));
    while (factor <= half) {
        if (n % factor == 0) {
            int f1 = factor;
            int score = get_score(f1);
            if (score > best_score ||
                (score == best_score && f1 < best_factor)) {
                best_factor = f1;
                best_score = score;
            }
            int f2 = n / factor;
            score = get_score(f2);
            if (score > best_score ||
                (score == best_score && f2 < best_factor)) {
                best_factor = f2;
                best_score = score;
            }
        }
        factor++;
    }
    cout << best_factor << endl;
    return 0;
}
