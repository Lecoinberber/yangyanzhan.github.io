solution: 

int main() {
    int T;
    cin >> T;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    for (int t = 0; t < T; t++) {
        string nk_temp;
        getline(cin, nk_temp);
        vector<string> nk = split_string(nk_temp);
        int n = stoi(nk[0]);
        int k = stoi(nk[1]);
        int res = 0;
        if (k < n / 2) {
            res = 2 * (k + 1) - 1;
        } else {
            res = 2 * (n - 1 - k);
        }
        cout << res << endl;
    }
    return 0;
}
