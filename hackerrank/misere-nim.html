solution: 

string misereNim(vector<int> s) {
    if (s.size() == 1) {
        return s[0] > 1 ? "First" : "Second";
    }
    int res = 0, sum = 0;
    for (auto num : s) {
        res = res ^ num;
        sum += num;
    }
    if (sum == s.size()) {
        return s.size() % 2 == 0 ? "First" : "Second";
    }
    return res == 0 ? "Second" : "First";
}
