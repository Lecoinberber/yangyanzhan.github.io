solution: 

DoublyLinkedListNode *reverse(DoublyLinkedListNode *head) {
    auto curr = head;
    while (curr->next) {
        auto next = curr->next;
        auto nnext = next->next;
        next->next = head;
        head->prev = next;
        head = next;
        curr->next = nnext;
        if (nnext) {
            nnext->prev = curr;
        }
    }
    return head;
}
