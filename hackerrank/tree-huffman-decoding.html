solution: 

void decode_huff(node *root, string s) {
    auto curr = root;
    int i = 0;
    vector<char> res;
    while (i < s.size()) {
        auto ch = s[i];
        if (ch == '0') {
            if (curr->left) {
                curr = curr->left;
                i++;
            } else {
                res.push_back(curr->data);
                curr = root;
            }
        } else {
            if (curr->right) {
                curr = curr->right;
                i++;
            } else {
                res.push_back(curr->data);
                curr = root;
            }
        }
    }
    res.push_back(curr->data);
    cout << string(res.begin(), res.end());
}
