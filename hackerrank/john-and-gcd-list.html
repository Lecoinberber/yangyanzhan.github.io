solution: 

int gcd(int a, int b) {
    if (a < b) {
        swap(a, b);
    }
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
int lcm(int a, int b) { return a / gcd(a, b) * b; }
vector<int> solve(vector nums) {
    vector<int> res{nums[0]};
    for (int i = 0; i < nums.size() - 1; i++) {
        int a = res[i], b = nums[i + 1], t = nums[i];
        int c = lcm(b, t);
        res.push_back(c);
    }
    res.push_back(nums.back());
    return res;
}
