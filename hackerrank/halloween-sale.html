solution: 

int howManyGames(long long p, long long d, long long m, long long s) {
    if (s < p) {
        return 0;
    }
    long long c1 = (p - m) / d + 1;
    long long t1 = (p + (p - (c1 - 1) * d)) * c1 / 2;
    if (s <= t1) {
        long long total = 0, count = 0;
        for (int i = 1; i <= c1; i++) {
            total += p;
            if (total > s) {
                break;
            }
            count++;
            p -= d;
            p = max(p, m);
        }
        return count;
    } else {
        return c1 + (s - t1) / m;
    }
}
