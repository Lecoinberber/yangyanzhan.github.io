solution: 

int birthday(vector<int> s, int d, int m) {
    if (s.size() < m) {
        return 0;
    }
    long long total = 0;
    for (int i = 0; i < m; i++) {
        total += s[i];
    }
    int res = 0;
    if (total == d) {
        res++;
    }
    for (int i = m; i < s.size(); i++) {
        total -= s[i - m];
        total += s[i];
        if (total == d) {
            res++;
        }
    }
    return res;
}
