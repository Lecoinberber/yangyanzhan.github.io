solution: 

int gcd(int a, int b) {
    if (b % a == 0) {
        return a;
    }
    return gcd(b % a, a);
}
string solve(vector<int> a) {
    int n = a.size(), g = a[0];
    for (int i = 1; i < n; i++) {
        int a1 = min(g, a[i]);
        int a2 = max(g, a[i]);
        g = gcd(a1, a2);
        if (g == 1) {
            break;
        }
    }
    return g == 1 ? "YES" : "NO";
}
