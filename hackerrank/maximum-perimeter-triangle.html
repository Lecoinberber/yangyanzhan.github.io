solution: 

vector<int> maximumPerimeterTriangle(vector sticks) {
    sort(sticks.begin(), sticks.end());
    for (int i = sticks.size() - 1; i >= 2; i--) {
        for (int j = i - 1; j >= 1; j--) {
            for (int k = j - 1; k >= 0; k--) {
                int a = sticks[i], b = sticks[j], c = sticks[k];
                if (b + c <= a) {
                    continue;
                }
                return vector<int>{c, b, a};
            }
        }
    }
    return vector<int>{-1};
}
