solution: 

long strangeCounter(long long t) {
    long long n = ceil(log2(t / 3.0 + 1));
    long long start = 1;
    for (int i = 2; i <= n; i++) {
        start *= 2;
    }
    start *= 3;
    long long pre = 0;
    for (long long m = 1; m < n; m++) {
        long long start = 1;
        for (int i = 2; i <= m; i++) {
            start *= 2;
        }
        start *= 3;
        pre += start;
    }
    t -= pre;
    return start - t + 1;
}
