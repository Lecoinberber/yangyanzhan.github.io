solution: 

void minimumBribes(vector<int> q) {
    int res = 0;
    for (int i = 0; i < q.size(); i++) {
        int num = q[i];
        if ((num - 1 - i) > 2) {
            cout << "Too chaotic" << endl;
            return;
        }
        for (int j = max(0, num - 1 - 2); j < i; j++) {
            if (q[j] > num) {
                res++;
            }
        }
    }
    cout << res << endl;
}
