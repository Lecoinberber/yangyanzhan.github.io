solution: 

string solve(int n) {
    unsigned long long trial = 1;
    while (true) {
        unsigned long long number = 0;
        unsigned long long tmp = trial;
        unsigned long long base = 1;
        while (tmp > 0) {
            number += 9 * (tmp % 2) * base;
            tmp /= 2;
            base *= 10;
        }
        if (number % n == 0) {
            return to_string(number);
        }
        trial++;
    }
}
