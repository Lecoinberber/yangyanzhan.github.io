solution: 

string isValid(string s) {
    map<char, int> mapping;
    for (auto ch : s) {
        if (mapping.find(ch) == mapping.end()) {
            mapping[ch] = 0;
        }
        mapping[ch]++;
    }
    int num1, num2, cnt1 = 0, cnt2 = 0;
    for (auto pair : mapping) {
        int num = pair.second;
        if (cnt1 == 0) {
            num1 = num;
            cnt1++;
        } else {
            if (num1 == num) {
                cnt1++;
            } else if (cnt2 == 0) {
                num2 = num;
                cnt2++;
            } else {
                if (num2 == num) {
                    if (cnt1 == 1) {
                        swap(num1, num2);
                        swap(cnt1, cnt2);
                        cnt1++;
                    } else {
                        return "NO";
                    }
                } else {
                    return "NO";
                }
            }
        }
    }
    return (cnt2 == 0 || num2 == 1 || (cnt1 == 1 && abs(num1 - num2) == 1) ||
            num2 - num1 == 1)
               ? "YES"
               : "NO";
}
