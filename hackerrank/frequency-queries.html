solution: 

vector<int> freqQuery(vector> queries) {
    vector<int> res;
    map<int, int> mapping;
    map<int, int> counts;
    for (auto query : queries) {
        switch (query[0]) {
        case 1: {
            int num = query[1], count;
            if (mapping.find(num) == mapping.end()) {
                mapping[num] = 1;
                count = 1;
                if (counts.find(count) == counts.end()) {
                    counts[count] = 1;
                } else {
                    counts[count]++;
                }
            } else {
                count = mapping[num];
                counts[count]--;
                mapping[num]++;
                count++;
                if (counts.find(count) == counts.end()) {
                    counts[count] = 1;
                } else {
                    counts[count]++;
                }
            }
            break;
        }
        case 2: {
            int num = query[1], count;
            if (mapping.find(num) != mapping.end() &&
                (count = mapping[num]) > 0) {
                counts[count]--;
                mapping[num]--;
                count--;
                if (count > 0) {
                    if (counts.find(count) == counts.end()) {
                        counts[count] = 1;
                    } else {
                        counts[count]++;
                    }
                }
            }
            break;
        }
        case 3: {
            int count = query[1];
            bool found =
                counts.find(count) != counts.end() && counts[count] > 0;
            res.push_back(found ? 1 : 0);
            break;
        }
        }
    }
    return res;
}
