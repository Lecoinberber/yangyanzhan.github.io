solution: 

unsigned long long solve(int n, vector<vector> operations) {
    unsigned long long total = 0;
    for (auto operation : operations) {
        unsigned long long a = operation[0], b = operation[1], c = operation[2];
        total += (b - a + 1) * c;
    }
    return total / n;
}
