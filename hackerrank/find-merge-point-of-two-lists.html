solution: 

int findMergeNode(SinglyLinkedListNode *head1, SinglyLinkedListNode *head2) {
    auto curr1 = head1, curr2 = head2;
    while (curr1 != curr2) {
        if (curr1->next) {
            curr1 = curr1->next;
        } else {
            curr1 = head2;
        }
        if (curr2->next) {
            curr2 = curr2->next;
        } else {
            curr2 = head1;
        }
    }
    return curr1;
}
