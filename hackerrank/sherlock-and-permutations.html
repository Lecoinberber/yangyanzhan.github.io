solution: 

int gcd(int m, int n, int &x, int &y) {
    if (n % m == 0) {
        x = 1;
        y = 0;
        return m;
    }
    int x1, y1;
    int res = gcd(n % m, m, x1, y1);
    y = x1;
    x = y1 - n / m * x1;
    return res;
}
int solve(int n, int m) {
    int a = min(m - 1, n), b = max(m - 1, n), c = m + n - 1;
    unsigned long long res = 1;
    unsigned long long base = 1000000007;
    for (int i = b + 1; i <= c; i++) {
        res *= (i % base);
        res %= base;
    }
    for (int i = 2; i <= a; i++) {
        int x, y;
        gcd(i, base, x, y);
        x = (x + base) % base;
        res *= x;
        res %= base;
    }
    return res;
}
