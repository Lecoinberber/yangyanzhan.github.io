solution: 

int luckBalance(int k, vector<vector> contests) {
    vector<int> imps;
    int res = 0;
    for (auto contest : contests) {
        bool important = contest[1] == 1;
        int luck = contest[0];
        if (!important) {
            res += luck;
        } else {
            imps.push_back(luck);
        }
    }
    sort(imps.begin(), imps.end());
    reverse(imps.begin(), imps.end());
    for (int i = 0; i < imps.size(); i++) {
        res += i < k ? imps[i] : -imps[i];
    }
    return res;
}
