solution: 

int flatlandSpaceStations(int n, vector<int> c) {
    sort(c.begin(), c.end());
    int res = 0;
    for (int i = 0; i < c.size() - 1; i++) {
        int dist = (c[i + 1] - c[i]) / 2;
        res = max(res, dist);
    }
    res = max(res, c[0]);
    res = max(res, n - 1 - c[c.size() - 1]);
    return res;
}
