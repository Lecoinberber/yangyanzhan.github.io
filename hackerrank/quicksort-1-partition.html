solution: 

vector<int> quickSort(vector arr) {
    int pivot = arr[0];
    vector<int> lefts, rights;
    for (int i = 1; i < arr.size(); i++) {
        if (arr[i] <= pivot) {
            lefts.push_back(arr[i]);
        } else {
            rights.push_back(arr[i]);
        }
    }
    lefts.push_back(pivot);
    lefts.insert(lefts.end(), rights.begin(), rights.end());
    return lefts;
}
