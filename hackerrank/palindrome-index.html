solution: 

int palindromeIndex(string s) {
    int begin = 0, end = s.size() - 1;
    while (begin < end) {
        if (s[begin] != s[end]) {
            int ori_begin = begin, ori_end = end;
            begin++;
            while (begin < end) {
                if (s[begin] != s[end]) {
                    break;
                }
                begin++;
                end--;
            }
            if (begin >= end) {
                return ori_begin;
            }
            begin = ori_begin;
            end = ori_end - 1;
            while (begin < end) {
                if (s[begin] != s[end]) {
                    break;
                }
                begin++;
                end--;
            }
            if (begin >= end) {
                return ori_end;
            }
            break;
        }
        begin++;
        end--;
    }
    return -1;
}
