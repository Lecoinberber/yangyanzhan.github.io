solution: 

vector<int> breakingRecords(vector scores) {
    int max_score = scores[0], min_score = scores[0], max_count = 0,
        min_count = 0;
    for (int i = 1; i < scores.size(); i++) {
        int score = scores[i];
        if (score > max_score) {
            max_score = score;
            max_count++;
        }
        if (score < min_score) {
            min_score = score;
            min_count++;
        }
    }
    return {max_count, min_count};
}
