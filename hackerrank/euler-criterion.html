solution: 

int pow_mod(int a, int b, int m) {
    if (b == 0) {
        return 1;
    }
    long long res = pow_mod(a, b / 2, m);
    res *= res;
    res %= m;
    if (b % 2 == 1) {
        res *= a;
        res %= m;
    }
    return res;
}
string solve(int a, int m) {
    int res = pow_mod(a, (m - 1) / 2, m);
    if (res == 1 || a == 0 || a == 2) {
        return "YES";
    }
    return "NO";
}
