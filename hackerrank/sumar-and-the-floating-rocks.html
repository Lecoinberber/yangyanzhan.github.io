solution: 

int gcd(int a, int b) {
    if (b % a == 0) {
        return a;
    }
    return gcd(b % a, a);
}
int solve(int x1, int y1, int x2, int y2) {
    if (x1 == x2) {
        return abs(y1 - y2) - 1;
    }
    if (y1 == y2) {
        return abs(x1 - x2) - 1;
    }
    int a = min(abs(x1 - x2), abs(y1 - y2));
    int b = max(abs(x1 - x2), abs(y1 - y2));
    return gcd(a, b) - 1;
}
