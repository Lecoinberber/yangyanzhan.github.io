solution: 

string catAndMouse(int x, int y, int z) {
    if (x == y) {
        return "Mouse C";
    }
    if (z <= min(x, y)) {
        return x < y ? "Cat A" : "Cat B";
    }
    if (z >= max(x, y)) {
        return x > y ? "Cat A" : "Cat B";
    }
    if (2 * z == x + y) {
        return "Mouse C";
    }
    int diff1 = abs(x - z), diff2 = abs(y - z);
    return diff1 < diff2 ? "Cat A" : "Cat B";
}
