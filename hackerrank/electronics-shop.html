solution: 

int getMoneySpent(vector<int> keyboards, vector drives, int b) {
    int max_sum = -1;
    for (auto keyboard : keyboards) {
        for (auto drive : drives) {
            long long sum = (long long)keyboard + drive;
            if (sum <= b) {
                max_sum = max(max_sum, (int)sum);
            }
        }
    }
    return max_sum;
}
