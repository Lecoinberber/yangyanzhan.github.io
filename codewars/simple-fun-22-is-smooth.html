solution: 

using namespace std;
#include <vector>
bool isSmooth(const std::vector<int> &arr) {
    int n = arr.size();
    int n1 = arr[0];
    int n2 = arr[n - 1];
    int n3;
    if (n % 2 != 0) {
        n3 = arr[n / 2];
    } else {
        n3 = arr[n / 2 - 1] + arr[n / 2];
    }
    return n1 == n3 && n3 == n2;
}
