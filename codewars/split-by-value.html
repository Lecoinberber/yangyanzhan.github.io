solution: 

using namespace std;
std::vector<int> split_by_value(int k, std::vector elements) {
    vector<int> r1, r2;
    for (auto num : elements) {
        if (num < k) {
            r1.push_back(num);
        } else {
            r2.push_back(num);
        }
    }
    vector<int> r;
    for (auto num : r1) {
        r.push_back(num);
    }
    for (auto num : r2) {
        r.push_back(num);
    }
    return r;
}
