solution: 

#include <bitset>
#include <string>
#include <utility>
using namespace std;
std::pair<int, int> splitNumbers(int n) {
    auto item = bitset<32>(n).to_string();
    auto item1 = item, item2 = item;
    int c = 0;
    for (int i = item.size() - 1; i >= 0; i--) {
        if (item[i] == '1') {
            c++;
            if (c % 2 == 1) {
                item2[i] = '0';
            } else {
                item1[i] = '0';
            }
        }
    }
    return {stoi(item1, nullptr, 2), stoi(item2, nullptr, 2)};
}
