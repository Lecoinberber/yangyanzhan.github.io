solution: 

#include <sstream>
#include <string>
using namespace std;
std::string womens_age(unsigned n) {
    int base, age;
    if (n % 2 == 0) {
        base = n / 2;
        age = 20;
    } else {
        base = (n - 1) / 2;
        age = 21;
    }
    stringstream ss;
    ss << n << "? That's just " << age << ", in base " << base << "!";
    return ss.str();
}
