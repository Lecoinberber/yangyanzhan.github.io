solution: 

class Pattern {
  public:
    static string pattern(int n);
};
string Pattern::pattern(int n) {
    using namespace std;
    if (n <= 0) {
        return "";
    }
    string res;
    for (int i = 0; i < n; i++) {
        string item;
        for (int j = n; j >= n - i; j--) {
            item += to_string(j);
        }
        res += item;
        if (i != n - 1) {
            res += "\n";
        }
    }
    return res;
}
