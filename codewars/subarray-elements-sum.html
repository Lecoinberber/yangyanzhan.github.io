solution: 

#include <vector>
using namespace std;
long elementsSum(const std::vector<std::vector> &arr, int d = 0) {
    int n = arr.size();
    long res = 0;
    for (int i = 0; i < n; i++) {
        int pos = n - 1 - i;
        if (pos >= arr[i].size()) {
            res += d;
        } else {
            res += arr[i][pos];
        }
    }
    return res;
}
