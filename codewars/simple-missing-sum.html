solution: 

using namespace std;
#include <set>
set<long long> sum_up(vector nums) {
    set<long long> res{0};
    if (nums.size() <= 0) {
        return res;
    }
    long long num = nums[0];
    res.insert(num);
    vector<int> sub_nums(nums.begin() + 1, nums.end());
    auto sub_sum = sum_up(sub_nums);
    for (auto n : sub_sum) {
        res.insert(n);
        res.insert(n + num);
    }
    return res;
}
int solve(std::vector<int> v) {
    set<long long> nums = sum_up(v);
    int res = 1;
    while (true) {
        if (nums.find(res) == nums.end()) {
            return res;
        }
        res++;
    }
    return -1;
}
