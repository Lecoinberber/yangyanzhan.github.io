solution: 

#include <algorithm>
#include <cmath>
#include <utility>
#include <vector>
using vec_rect_circle = std::vector<std::pair>;
using namespace std;
double area(pair<double, double> xy) {
    auto x = get<0>(xy);
    auto y = get<1>(xy);
    if (y == 0) {
        return M_PI * x * x;
    } else {
        return x * y;
    }
}
vec_rect_circle sort_by_area(vec_rect_circle v) {
    sort(v.begin(), v.end(),
         [](pair<double, double> a, pair b) {
             return area(a) < area(b);
         });
    return v;
}
