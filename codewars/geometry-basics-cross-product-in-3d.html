solution: 

#include <vector>
using namespace std;
Vector cross_product(const Vector &a, const Vector &b) {
    auto a1 = a.x, a2 = a.y, a3 = a.z;
    auto b1 = b.x, b2 = b.y, b3 = b.z;
    auto c1 = a2 * b3 - a3 * b2;
    auto c2 = -(a1 * b3 - a3 * b1);
    auto c3 = a1 * b2 - a2 * b1;
    return {c1, c2, c3};
}
