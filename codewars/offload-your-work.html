solution: 

#include <sstream>
using namespace std;
std::string workNeeded(int projectMinutes,
                       std::vector<std::pair> freelancers) {
    int p = projectMinutes;
    for (auto lancer : freelancers) {
        int m = get<0>(lancer) * 60 + get<1>(lancer);
        if (p <= m) {
            return "Easy Money!";
        }
        p -= m;
    }
    stringstream ss;
    int h = p / 60;
    int m = p % 60;
    ss << "I need to work " << h << " hour(s) and " << m << " minute(s)";
    return ss.str();
}
