solution: 

using namespace std;
std::string solve(std::string s, int a, int b) {
    int n = s.size();
    a = max(0, a);
    a = min(a, n - 1);
    b = max(0, b);
    b = min(b, n - 1);
    b = max(a, b);
    reverse(s.begin() + a, s.begin() + b + 1);
    return s;
}
