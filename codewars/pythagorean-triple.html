solution: 

bool PythagoreanTriple(const int a, const int b, const int c) {
    using namespace std;
    vector<int> nums{a, b, c};
    sort(nums.begin(), nums.end());
    int n1 = nums[0], n2 = nums[1], n3 = nums[2];
    return n1 * n1 + n2 * n2 == n3 * n3;
}
