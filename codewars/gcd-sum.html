solution: 

using namespace std;
int gcd(int a, int b) {
    if (a < b) {
        return gcd(b, a);
    }
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
pair<int, int> solve(int s, int g) {
    if (s % g != 0) {
        return {-1, -1};
    }
    int p = s / g;
    for (int i = 1; i <= p / 2; i++) {
        if (gcd(i, p - i) == 1) {
            return {i * g, (p - i) * g};
        }
    }
    return {-1, -1};
}
