solution: 

#include <set>
std::vector<int> testit(std::vector a, std::vector b) {
    using namespace std;
    set<int> a1(a.begin(), a.end());
    set<int> b1(b.begin(), b.end());
    vector<int> res(a1.begin(), a1.end());
    res.insert(res.end(), b1.begin(), b1.end());
    sort(res.begin(), res.end());
    return res;
}
