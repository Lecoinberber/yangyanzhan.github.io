solution: 

#include <cmath>
#include <ctime>
using namespace std;
class DateDays {
  public:
    static std::string dateNbDays(double a0, double a, double p) {
        int day = ceil(log(a / a0) / log(p / 36000 + 1));
        tm time = {};
        strptime("2016-01-01", "%Y-%m-%d", &time);
        time.tm_mday += day;
        mktime(&time);
        char buffer[100];
        strftime(buffer, 100, "%Y-%m-%d", &time);
        return string(buffer);
    }
};
