solution: 

#include <chrono>
using namespace std;
int timed_reading(int maxLength, const std::string &text) {
    int res = 0;
    int pre = -1;
    for (int i = 0; i < text.size(); i++) {
        if (isalpha(text[i])) {
            if (pre == -1) {
                pre = i;
            }
        }
        if (pre != -1) {
            int len = 0;
            if (!isalpha(text[i])) {
                len = i - pre;
            } else if (i == text.size() - 1) {
                len = i - pre + 1;
            }
            if (len > 0) {
                if (len <= maxLength) {
                    res++;
                }
                pre = -1;
            }
        }
    }
    return res;
}
