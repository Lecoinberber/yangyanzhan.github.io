solution: 

using namespace std;
bool compare(std::string s1, std::string s2) {
    int sum1 = 0;
    for (auto ch : s1) {
        if (!isalpha(ch)) {
            sum1 = 0;
            break;
        }
        ch = toupper(ch);
        sum1 += ch;
    }
    int sum2 = 0;
    for (auto ch : s2) {
        if (!isalpha(ch)) {
            sum2 = 0;
            break;
        }
        ch = toupper(ch);
        sum2 += ch;
    }
    return sum1 == sum2;
}
