solution: 

using namespace std;
#include <set>
#include <string>
unsigned int countConsonants(const std::string &str) {
    set<char> vowels = {'a', 'e', 'i', 'o', 'u'};
    int res = 0;
    vector<int> cnts(26, 0);
    for (auto ch : str) {
        ch = tolower(ch);
        int pos = ch - 'a';
        if (0 <= pos && pos < 26) {
            cnts[pos]++;
        }
    }
    for (int i = 0; i < 26; i++) {
        char ch = 'a' + i;
        if (vowels.find(ch) == vowels.end() && cnts[i] >= 1) {
            res++;
        }
    }
    return res;
}
