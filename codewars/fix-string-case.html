solution: 

#include <string>
std::string solve(const std::string &str) {
    using namespace std;
    int c1 = 0, c2 = 0;
    for (auto &ch : str) {
        if (tolower(ch) == ch) {
            c1++;
        } else {
            c2++;
        }
    }
    string res;
    for (auto &ch : str) {
        if (c1 >= c2) {
            res += tolower(ch);
        } else {
            res += toupper(ch);
        }
    }
    return res;
}
