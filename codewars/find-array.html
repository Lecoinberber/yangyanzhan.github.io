solution: 

using namespace std;
template <typename T> vector find_array(vector arr1, vector arr2) {
    vector<T> res;
    for (auto idx : arr2) {
        if (0 <= idx && idx < arr1.size()) {
            res.push_back(arr1[idx]);
        }
    }
    return res;
}
