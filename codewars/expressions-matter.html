solution: 

unsigned short int expressionsMatter(unsigned short int a, unsigned short int b,
                                     unsigned short int c) {
    using namespace std;
    int r1 = a * (b + c);
    int r2 = (a + b) * c;
    int r3 = a * b + c;
    int r4 = a + b * c;
    int r5 = a + b + c;
    int r6 = a * b * c;
    vector<int> rs{r1, r2, r3, r4, r5, r6};
    return *max_element(rs.begin(), rs.end());
}
