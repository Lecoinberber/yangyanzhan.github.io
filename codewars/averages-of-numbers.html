solution: 

std::vector<double> averages(std::vector numbers) {
    using namespace std;
    vector<double> res;
    for (int i = 1; i < numbers.size(); i++) {
        res.push_back(((double)numbers[i - 1] + numbers[i]) / 2);
    }
    return res;
}
