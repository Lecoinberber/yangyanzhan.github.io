using namespace std;
std::string wordPattern(const std::string &word) {
    string res;
    char pre = '\0';
    int c = 0;
    vector<int> rs(26, -1);
    for (auto ch : word) {
        ch = tolower(ch);
        int pos = ch - 'a';
        if (rs[pos] == -1) {
            rs[pos] = c;
            c++;
        }
        res += to_string(rs[pos]) + ".";
    }
    res = res.substr(0, res.size() - 1);
    return res;
}
