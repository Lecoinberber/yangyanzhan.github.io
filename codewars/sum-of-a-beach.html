solution: 

using namespace std;
int sum_of_a_beach(string s) {
    vector<string> words = {"sand", "water", "fish", "sun"};
    for (auto &ch : s) {
        ch = tolower(ch);
    }
    int res = 0;
    int idx = 0;
    while (idx < s.size()) {
        bool found = false;
        for (auto word : words) {
            int idx1 = s.find(word, idx);
            if (idx1 == idx) {
                found = true;
                idx += word.size();
                res++;
                break;
            }
        }
        if (!found) {
            idx++;
        }
    }
    return res;
}
