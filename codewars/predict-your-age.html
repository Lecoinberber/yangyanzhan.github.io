solution: 

using namespace std;
#include <math.h>
int predictAge(int age1, int age2, int age3, int age4, int age5, int age6,
               int age7, int age8) {
    vector<int> ages = {age1, age2, age3, age4, age5, age6, age7, age8};
    int res = 0;
    for (auto age : ages) {
        res += age * age;
    }
    res = sqrt(res);
    res /= 2;
    return res;
}
