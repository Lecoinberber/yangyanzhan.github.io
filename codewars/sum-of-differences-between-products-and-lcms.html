solution: 

#include <vector>
using namespace std;
int gcd(int a, int b) {
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
int lcm(int a, int b) {
    int g = gcd(max(a, b), min(a, b));
    return a / g * b;
}
int sum_differences_between_products_and_LCMs(
    const vector<vector> &pairs) {
    int res = 0;
    for (auto pair : pairs) {
        int a = pair[0], b = pair[1];
        if (a != 0 && b != 0) {
            int g = gcd(max(a, b), min(a, b));
            res += a * b - a * b / g;
        }
    }
    return res;
}
