solution: 

#include <cmath>
#include <sstream>
using namespace std;
std::string draw_spider(int legSize, int bodySize, char mouth, char eye) {
    vector<vector> legs = {
        {"^", "^"}, {"/\\", "/\\"}, {"/╲", "╱\\"}, {"╱╲", "╱╲"}};
    stringstream ss;
    ss << legs[legSize - 1][0];
    for (int i = 0; i < bodySize; i++) {
        ss << "(";
    }
    for (int i = 0; i < pow(2, bodySize) / 2; i++) {
        ss << (char)eye;
    }
    ss << (char)mouth;
    for (int i = 0; i < pow(2, bodySize) / 2; i++) {
        ss << (char)eye;
    }
    for (int i = 0; i < bodySize; i++) {
        ss << ")";
    }
    ss << legs[legSize - 1][1];
    return ss.str();
}
