solution: 

using namespace std;
string t_b = "Begin on ";
string t_r = "Right on ";
string t_l = "Left on ";
vector<string> extract(string item) {
    vector<string> tokens = {t_b, t_r, t_l};
    int idx = string::npos;
    for (auto token : tokens) {
        idx = item.find(token);
        if (idx != string::npos) {
            return {token, item.substr(idx + token.size())};
        }
    }
    return {};
}
std::vector<std::string> solve(std::vector v) {
    vector<vector> items;
    vector<string> res;
    string pre_dir = t_b;
    for (int i = v.size() - 1; i >= 0; i--) {
        auto items = extract(v[i]);
        string dir = t_b;
        if (pre_dir == t_r) {
            dir = t_l;
        } else if (pre_dir == t_l) {
            dir = t_r;
        }
        res.push_back(dir + items[1]);
        pre_dir = items[0];
    }
    return res;
}
