solution: 

class Solution {
  public:
    static float money;
    static float candy;
    static float chips;
    static float soda;
    static float change;
};
float Solution::money = 10;
float Solution::candy = 1.42;
float Solution::chips = 2.40;
float Solution::soda = 1;
float Solution::change =
    Solution::money - Solution::candy - Solution::chips - Solution::soda;
