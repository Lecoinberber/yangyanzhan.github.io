solution: 

#include <algorithm>
#include <string>
bool isAnagram(std::string test, std::string original) {
    using namespace std;
    string a = test;
    for (auto &ch : a) {
        ch = tolower(ch);
    }
    string b = original;
    for (auto &ch : b) {
        ch = tolower(ch);
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    return a == b;
}
