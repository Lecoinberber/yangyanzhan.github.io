my @rows = [1],;
sub next-row($n) {
    my $row = [];
    for 1..$n -> $i {
        $row.push([+](@rows[$n - $i][0..^min($i, @rows[$n - $i].elems)]));
    }
    @rows.push($row);
    @rows[$n];
}
say "rows:";
say .fmt('%2d'), ": ", next-row($_) for 1..10;
