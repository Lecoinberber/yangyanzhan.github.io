solution: 

sub solution1() {
    my @doors = False xx 101;
    (.=not for @doors[0, $_ ... 100]) for 1..100;
    say "Door $_ is ", <closed open>[ @doors[$_] ] for 1..100;
}
sub solution2() {
    say "Door $_ is open" for map {$^n ** 2}, 1..10;
}
sub solution3() {
    say "Door $_ is ", <closed open>[.sqrt == .sqrt.floor] for 1..100;
}
solution1;
solution2;
solution3;
