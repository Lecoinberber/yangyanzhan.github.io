solution: 

#include <iostream>
using namespace std;
int main() {
    string item;
    cin >> item;
    int cache[10] = {0};
    for (auto ch : item) {
        cache[ch - '0']++;
    }
    bool balanced = true;
    for (int i = 1; i < 10; i++) {
        if (cache[i] != cache[i - 1]) {
            balanced = false;
        }
    }
    cout << (balanced ? 1 : 0);
    return 0;
}
