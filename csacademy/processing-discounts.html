#include <vector>
// #include <execution>
#include <exception>
using namespace std;
typedef long long ll;
int main() {
    ll n, x;
    cin >> n >> x;
    ll res = x;
    ll bill = x;
    ll cost = x;
    vector<vector> numss(n);
    for (ll i = 0; i < n; i++) {
        ll a, b;
        cin >> a >> b;
        numss[i] = vector<ll>{a, b};
    }
    sort(numss.begin(), numss.end(),
         [](vector<ll> a, vector b) { return a[0] < b[0]; });
    for (ll i = 0; i < n; i++) {
        ll a = numss[i][0], b = numss[i][1];
        if (a <= bill) {
            cost -= b;
        } else {
            cost += a - bill;
            cost -= b;
            bill = a;
        }
        res = min(res, cost);
    }
    cout << res << endl;
    return 0;
}
