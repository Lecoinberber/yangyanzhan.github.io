#include <exception>
using namespace std;
typedef long long ll;
int main() {
    ios::sync_with_stdio(false);
    ll n, c, x1, u, x2;
    cin >> n >> c >> x1 >> u >> x2;
    ll res = c;
    ll inc = x1;
    for (int i = 1; i <= n; i++) {
        if (res >= u) {
            if ((n - i + 1) * x2 > u) {
                res -= u;
                inc += x2;
            } else {
                res += inc * (n - i + 1);
                break;
            }
        }
        res += inc;
    }
    cout << res << endl;
    return 0;
}
