#include <algorithm>

#include <utility>
#include <vector>
// #include <execution>
#include <exception>
using namespace std;
typedef long long ll;
int main() {
    int n;
    cin >> n;
    map<int, set> mapping1, mapping2;
    for (int i = 0; i < n - 1; i++) {
        int a, b;
        cin >> a >> b;
        if (mapping1.find(a) == mapping1.end()) {
            mapping1[a] = set<int>();
        }
        mapping1[a].insert(b);
        swap(a, b);
        if (mapping1.find(a) == mapping1.end()) {
            mapping1[a] = set<int>();
        }
        mapping1[a].insert(b);
    }
    for (int i = 0; i < n - 1; i++) {
        int a, b;
        cin >> a >> b;
        if (mapping2.find(a) == mapping2.end()) {
            mapping2[a] = set<int>();
        }
        mapping2[a].insert(b);
        swap(a, b);
        if (mapping2.find(a) == mapping2.end()) {
            mapping2[a] = set<int>();
        }
        mapping2[a].insert(b);
    }
    int res = 0;
    for (int i = 1; i <= n; i++) {
        if (mapping1.find(i) != mapping1.end()) {
            for (auto j : mapping1[i]) {
                if (mapping2.find(j) != mapping2.end() &&
                    mapping2[j].find(i) != mapping2[j].end()) {
                    res++;
                }
            }
        }
    }
    res /= 2;
    cout << res << endl;
    return 0;
}
