int main() {
    int n;
    cin >> n;
    int cache[4] = {0};
    for (int i = 0; i < n; i++) {
        int a, b;
        cin >> a >> b;
        if (a == 0 || b == 0) {
            continue;
        }
        a = a > 0 ? 1 : -1;
        b = b > 0 ? 1 : -1;
        int c;
        if (a * b < 0) {
            c = a > 0 ? 0 : 1;
        } else {
            c = a > 0 ? 2 : 3;
        }
        cache[c] = 1;
    }
    int res = 0;
    for (int i = 0; i < 4; i++) {
        res += cache[i];
    }
    cout << res << endl;
    return 0;
}
