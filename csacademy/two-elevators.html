#include <vector>
// #include <execution>
#include <exception>
using namespace std;
typedef long long ll;
int main() {
    int n, e1, e2;
    cin >> n >> e1 >> e2;
    vector<int> nums(2 * n);
    for (int i = 0; i < n; i++) {
        cin >> nums[2 * i] >> nums[2 * i + 1];
    }
    int r1 = 0, r2 = 0;
    for (int i = 0; i < n; i++) {
        int a = nums[2 * i], b = nums[2 * i + 1];
        if (e1 >= a && a >= b) {
            r1++;
        } else if (e2 <= a && a <= b) {
            r1++;
        }
        if (e1 <= a && a <= b) {
            r2++;
        } else if (e2 >= a && a >= b) {
            r2++;
        }
    }
    int res = max(r1, r2);
    cout << res << endl;
    return 0;
}
