description:

You are given an array AA of NN integers. Delete from the array all the elements equal to the minimum or the maximum value. Print the sum of the elements of the resulting array.

solution: 

#include <algorithm>
#include <bitset>
#include <chrono>
#include <cmath>
#include <cstdlib>
#include <deque>

#include <list>

#include <thread>
#include <tuple>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>
// #include <execution>
#include <exception>
using namespace std;
typedef long long ll;
int main() {
    ios::sync_with_stdio(false);
    ll n;
    cin >> n;
    vector<ll> nums(n);
    for (ll i = 0; i < n; i++) {
        cin >> nums[i];
    }
    auto pair = minmax_element(nums.begin(), nums.end());
    ll mini = *pair.first;
    ll maxi = *pair.second;
    ll res = 0;
    for (ll i = 0; i < n; i++) {
        auto num = nums[i];
        if (num != mini && num != maxi) {
            res += num;
        }
    }
    cout << res << endl;
    return 0;
}
