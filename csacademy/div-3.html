solution: 

#include <iostream>
using namespace std;
int main() {
    long long x;
    cin >> x;
    int res = 0;
    while (x != 1) {
        if (x == 2) {
            x--;
        } else if (x % 3 == 0) {
            x /= 3;
        } else if (x % 3 == 1) {
            x--;
        } else {
            x++;
        }
        res++;
    }
    cout << res;
    return 0;
}
