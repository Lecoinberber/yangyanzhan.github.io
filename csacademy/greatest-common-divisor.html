solution: 

#include <iostream>
using namespace std;
long long gcd(long long a, long long b) {
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
int main() {
    long long a, b;
    cin >> a >> b;
    cout << gcd(max(a, b), min(a, b));
    return 0;
}
