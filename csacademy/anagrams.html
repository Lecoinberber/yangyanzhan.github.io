solution: 

#include <algorithm>
#include <iostream>
#include <map>
#include <string>
using namespace std;
int main() {
    int n;
    cin >> n;
    map<string, int> cache;
    int res = 0;
    for (int i = 0; i < n; i++) {
        string item;
        cin >> item;
        sort(item.begin(), item.end());
        if (cache.find(item) == cache.end()) {
            cache[item] = 0;
        }
        cache[item]++;
        res = max(res, cache[item]);
    }
    cout << res;
    return 0;
}
