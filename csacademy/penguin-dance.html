using namespace std;
typedef long long ll;
int main() {
    int n, p, t;
    cin >> n >> p >> t;
    vector<int> offsets = {0, 0, 0, 0, 1, 0, 1, 2, 3};
    int c = t / offsets.size();
    int offset = c * 3;
    if (t % offsets.size() != 0) {
        offset += offsets[t % offsets.size() - 1];
    }
    int res = p - offset;
    if (res < 1 || res > n) {
        res = -1;
    }
    cout << res << endl;
    return 0;
}
