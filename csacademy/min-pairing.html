solution: 

#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
typedef long long ll;
int main() {
    ll n;
    cin >> n;
    vector<ll> nums;
    for (ll i = 0; i < n; i++) {
        ll num;
        cin >> num;
        nums.push_back(num);
    }
    sort(nums.begin(), nums.end());
    ll res = 0;
    for (ll i = 0; i < n; i += 2) {
        res += nums[i + 1] - nums[i];
    }
    cout << res;
    return 0;
}
