solution: 

std::vector<std::vector>>
financialCrisis(std::vector<std::vector> roadRegister) {
    using namespace std;
    int n = roadRegister.size();
    vector<vector>> res;
    for (int i = 0; i < n; i++) {
        vector<vector> graph;
        for (int i1 = 0; i1 < n; i1++) {
            if (i1 == i) {
                continue;
            }
            vector<bool> row;
            for (int i2 = 0; i2 < n; i2++) {
                if (i2 == i) {
                    continue;
                }
                row.push_back(roadRegister[i1][i2]);
            }
            graph.push_back(row);
        }
        res.push_back(graph);
    }
    return res;
}
