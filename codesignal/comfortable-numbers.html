solution: 

int getDigits(int n) {
    int res = 0;
    while (n > 0) {
        res += n % 10;
        n /= 10;
    }
    return res;
}
bool isComfort(int a, int b) {
    int digits = getDigits(a);
    return a != b && (a - digits <= b) && (b <= a + digits);
}
int comfortableNumbers(int L, int R) {
    int res = 0;
    for (int a = L; a < R; a++) {
        for (int b = a + 1; b <= R; b++) {
            if (isComfort(a, b) && isComfort(b, a)) {
                res++;
            }
        }
    }
    return res;
}
