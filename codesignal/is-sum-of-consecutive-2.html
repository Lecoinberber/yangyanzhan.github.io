solution: 

int isSumOfConsecutive2(int n) {
    int res = 0;
    for (int i = 1; i < n; i++) {
        int sum = i;
        for (int j = i + 1; j < n; j++) {
            sum += j;
            if (sum > n) {
                break;
            }
            if (sum == n) {
                res++;
            }
        }
    }
    return res;
}
