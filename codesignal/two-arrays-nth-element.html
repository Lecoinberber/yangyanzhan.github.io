solution: 

int twoArraysNthElement(std::vector<int> array1, std::vector array2,
                        int n) {
    using namespace std;
    int i1 = 0, i2 = 0, i = -1, n1 = array1.size(), n2 = array2.size(), res;
    while (i1 < n1 && i2 < n2 && i < n) {
        int num1 = array1[i1];
        int num2 = array2[i2];
        if (num1 <= num2) {
            i1++;
            res = num1;
        } else {
            i2++;
            res = num2;
        }
        i++;
    }
    while (i < n) {
        if (i1 < n1) {
            res = array1[i1];
            i1++;
            i++;
        } else if (i2 < n2) {
            res = array2[i2];
            i2++;
            i++;
        }
    }
    return res;
}
