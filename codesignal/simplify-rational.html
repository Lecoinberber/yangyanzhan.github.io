int gcd(int a, int b) {
    if (a < b) {
        return gcd(b, a);
    }
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
std::vector<int> simplifyRational(int numerator, int denominator) {
    using namespace std;
    if (numerator == 0) {
        return {0, 1};
    }
    int g = gcd(abs(numerator), abs(denominator));
    if (denominator < 0) {
        g = -g;
    }
    return {numerator / g, denominator / g};
}
