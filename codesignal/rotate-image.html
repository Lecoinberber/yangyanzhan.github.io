std::vector<std::vector> rotateImage(std::vector> nums) {
    using namespace std;
    int n = nums.size();
    for (int c = 1; c <= n / 2; c++) {
        for (int i = 1; i <= n - 2 * c + 1; i++) {
            swap(nums[c - 1][c - 1 + i - 1], nums[c - 1 + i - 1][n - c]);
        }
        for (int i = 1; i <= n - 2 * c + 1; i++) {
            swap(nums[c - 1][c - 1 + i - 1], nums[n - c][n - c - (i - 1)]);
        }
        for (int i = 1; i <= n - 2 * c + 1; i++) {
            swap(nums[c - 1][c - 1 + i - 1], nums[n - c - (i - 1)][c - 1]);
        }
    }
    return nums;
}
