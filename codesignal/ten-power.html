solution: 

int tens(int n) { return (n % 100) / 10; }
int TenPower(std::vector<int> numbers) {
    int res = 0, count = 0;
    for (int i = 0; i < numbers.size(); i++) {
        int n = numbers[i];
        if (n >= 10) {
            if (count % 2 == 0) {
                res += tens(n);
            } else {
                res -= tens(n);
            }
            count++;
        }
    }
    return res;
}
