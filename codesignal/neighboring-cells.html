solution: 

std::vector<std::vector>
neighboringCells(std::vector<std::vector> matrix) {
    using namespace std;
    int m = matrix.size(), n = matrix[0].size();
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (matrix[i][j] == 0) {
                int count = 4;
                if (m == 1 && n == 1) {
                    count = 0;
                }
                if (i == 0 || i == m - 1 || j == 0 || j == n - 1) {
                    count = 3;
                    if (m == 1 || n == 1) {
                        count = 2;
                    }
                    if ((i == 0 || i == m - 1) && (j == 0 || j == n - 1)) {
                        count = 2;
                        if (m == 1 || n == 1) {
                            count = 1;
                        }
                    }
                }
                matrix[i][j] = count;
            }
        }
    }
    return matrix;
}
