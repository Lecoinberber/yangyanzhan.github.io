solution: 

#include <algorithm>
std::string FibonacciSecret(std::string message) {
    using namespace std;
    string words;
    for (int i = 0; i < message.length(); i++) {
        if (!isspace(message[i])) {
            words += message[i];
        }
    }
    int a = 0, b = 1;
    string res;
    while (a < words.length()) {
        res += toupper(words[a]);
        res += "-";
        b = a + b;
        a = b - a;
    }
    if (res.length() > 0) {
        res = res.substr(0, res.length() - 1);
    }
    return res;
}
