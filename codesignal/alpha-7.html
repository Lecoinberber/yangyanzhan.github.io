solution: 

std::string alpha7(std::string s) {
    long num = 0;
    for (auto ch : s)
        num = 26 * num + ch - 97;
    std::string t = "";
    do {
        t = (char)(num % 7 + 48) + t;
    } while (num /= 7);
    return t;
}
