solution: 

bool checkSameElementExistence(std::vector<int> arr1, std::vector arr2) {
    int m = arr1.size(), n = arr2.size(), i = 0, j = 0;
    while (i < m && j < n) {
        if (arr1[i] == arr2[j]) {
            return true;
        } else if (arr1[i] < arr2[j]) {
            i++;
        } else {
            j++;
        }
    }
    return false;
}
