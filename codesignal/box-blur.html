std::vector<std::vector> boxBlur(std::vector> image) {
    using namespace std;
    int m = image.size(), n = image[0].size();
    vector<vector> res;
    for (int i = 1; i < m - 1; i++) {
        vector<int> row;
        for (int j = 1; j < n - 1; j++) {
            int sum = 0;
            for (int i1 = -1; i1 <= 1; i1++) {
                for (int j1 = -1; j1 <= 1; j1++) {
                    sum += image[i + i1][j + j1];
                }
            }
            row.push_back(sum / 9);
        }
        res.push_back(row);
    }
    return res;
}
