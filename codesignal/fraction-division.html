solution: 

int gcd(int a, int b) {
    if (a % b == 0) {
        return b;
    }
    return gcd(b, a % b);
}
std::vector<int> fractionDivision(std::vector a, std::vector b) {
    using namespace std;
    int a1 = a[0], a2 = a[1];
    int b1 = b[0], b2 = b[1];
    if (a1 == 0) {
        return {0, 1};
    }
    int c1 = a1 * b2, c2 = a2 * b1;
    int g = gcd(max(abs(c1), abs(c2)), min(abs(c1), abs(c2)));
    return {c1 / g, c2 / g};
}
