solution: 

int jumpingJimmy(std::vector<int> tower, int jumpHeight) {
    using namespace std;
    int h = 0, th = 0, i = -1, n = tower.size();
    while (true) {
        if (i >= n) {
            break;
        }
        int h1 = h + jumpHeight;
        int i1 = i + 1;
        for (; i1 < n; i1++) {
            if (h1 < th + tower[i1]) {
                break;
            }
            th += tower[i1];
        }
        i1--;
        if (i == i1) {
            break;
        }
        h = th;
        i = i1;
    }
    return h;
}
