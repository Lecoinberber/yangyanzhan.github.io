solution: 

std::vector<int> digitDifferenceSort(std::vector a) {
    using namespace std;
    vector<vector> pairs;
    int n = a.size();
    for (int i = 0; i < n; i++) {
        pairs.push_back({a[i], i});
    }
    function<int(int)> dd = [](int num) {
        if (num == 0) {
            return 0;
        }
        int min_d = num % 10, max_d = min_d;
        while (num > 0) {
            int d = num % 10;
            min_d = min(min_d, d);
            max_d = max(max_d, d);
            num /= 10;
        }
        return max_d - min_d;
    };
    sort(pairs.begin(), pairs.end(), [&](auto item1, auto item2) {
        int num1 = item1[0], idx1 = item1[1];
        int num2 = item2[0], idx2 = item2[1];
        int dd1 = dd(num1), dd2 = dd(num2);
        if (dd1 != dd2) {
            return dd1 < dd2;
        }
        return idx1 > idx2;
    });
    vector<int> res;
    for (int i = 0; i < n; i++) {
        res.push_back(pairs[i][0]);
    }
    return res;
}
