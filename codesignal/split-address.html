solution: 

#include <regex>
std::vector<std::string> splitAddress(std::string address) {
    using namespace std;
    regex reg("([a-z]+)://([a-z]+)\\.com(/[a-z]*)?");
    smatch matches;
    regex_search(address, matches, reg);
    vector<string> res;
    for (int i = 1; i < matches.size(); i++) {
        res.push_back(matches[i].str());
    }
    if (res[2].size() > 0) {
        res[2] = res[2].substr(1);
    } else {
        res.pop_back();
    }
    return res;
}
