solution: 

int comeOnDown(int maxPrice, std::vector<int> bids) {
    using namespace std;
    vector<int> nbids;
    for (auto bid : bids) {
        if (bid <= maxPrice) {
            nbids.push_back(bid);
        }
    }
    bids = nbids;
    sort(bids.begin(), bids.end());
    bids.insert(bids.begin(), 0);
    bids.push_back(maxPrice + 1);
    int res = 0, mwin = 0;
    for (int i = 0; i < bids.size() - 1; i++) {
        int a = bids[i] + 1;
        int b = bids[i + 1] - 1;
        if (b < a) {
            continue;
        }
        int win = b - a + 1;
        if (win > mwin) {
            res = a;
            mwin = win;
        }
    }
    return res;
}
