int multiDivisorSequence(std::vector<int> nums) {
    for (int i = 1; i < nums.size(); i++) {
        int now = nums[i], pre = nums[i - 1];
        if (pre % now == 0) {
            continue;
        } else {
            if (i == 1) {
            } else if (now % nums[i - 2] == 0) {
                continue;
            }
        }
        return now;
    }
    return -1;
}
