solution: 

int danceSteps(std::string danceFloor) {
    using namespace std;
    vector<bool> visited(danceFloor.size(), false);
    int pos = 0, step = 0;
    while (!visited[pos]) {
        visited[pos] = true;
        int gap = danceFloor[pos] - '0';
        int target = (gap % 2 == 0) ? (pos + gap) : (pos - gap);
        if (gap == 0) {
            return step;
        }
        if (0 <= target && target < danceFloor.size()) {
            step += gap;
            pos = target;
        } else if (target < 0) {
            step += pos;
            return step;
        } else {
            step += danceFloor.size() - 1 - pos;
            return step;
        }
    }
    return -1;
}
