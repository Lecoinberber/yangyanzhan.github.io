bool search(int n, std::vector<int> &fibs, std::vector &visited) {
    using namespace std;
    if (n == 0) {
        return true;
    }
    for (int i = fibs.size() - 1; i >= 0; i--) {
        if (fibs[i] <= n) {
            if (!visited[i]) {
                visited[i] = true;
                if (search(n - fibs[i], fibs, visited)) {
                    return true;
                }
                visited[i] = false;
            }
            if (i - 1 >= 0 && !visited[i - 1]) {
                visited[i - 1] = true;
                if (search(n - fibs[i - 1], fibs, visited)) {
                    return true;
                }
                visited[i - 1] = false;
            }
            break;
        }
    }
    return false;
}
std::vector<int> fibonacciSum(int n) {
    using namespace std;
    vector<int> fibs{1, 1};
    while (fibs.back() < n) {
        fibs.push_back(fibs[fibs.size() - 1] + fibs[fibs.size() - 2]);
    }
    vector<int> visited(fibs.size(), false);
    search(n, fibs, visited);
    vector<int> res;
    for (int i = 0; i < visited.size(); i++) {
        if (visited[i]) {
            res.push_back(fibs[i]);
        }
    }
    return res;
}
