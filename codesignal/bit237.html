solution: 

int get(int n, int k) {
    using namespace std;
    k--;
    int mask = 1 << k;
    return (n & mask) >> k;
}
int clear(int n, int k) {
    k--;
    int mask = ~(1 << k);
    return n & mask;
}
int set(int n, int k) {
    if (get(n, k)) {
        return n;
    }
    k--;
    return n + (1 << k);
}
int bit237(int n) {
    int s = get(n, 2) + get(n, 3) + get(n, 7);
    if (s <= 1) {
        n = clear(n, 2);
        n = clear(n, 3);
        n = clear(n, 7);
        return n;
    }
    n = set(n, 2);
    n = set(n, 3);
    n = set(n, 7);
    return n;
}
