solution: 

int getDigits(int num) {
    if (num == 0)
        return 1;
    int res = 0;
    while (num > 0) {
        res++;
        num /= 10;
    }
    return res;
}
int pagesNumberingWithInk(int current, int numberOfDigits) {
    int digits = getDigits(current);
    while (numberOfDigits >= digits) {
        current++;
        numberOfDigits -= digits;
        digits = getDigits(current);
    }
    return current - 1;
}
