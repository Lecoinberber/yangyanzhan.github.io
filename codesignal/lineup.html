solution: 

int lineUp(std::string commands) {
    using namespace std;
    int res = 0, a = 0, b = 0;
    for (auto ch : commands) {
        if (ch == 'L') {
            a--;
            b++;
        } else if (ch == 'R') {
            a++;
            b--;
        } else if (ch == 'A') {
            a += 2;
            b += 2;
        }
        a += 4;
        b += 4;
        a %= 4;
        b %= 4;
        if (a == b) {
            res++;
        }
    }
    return res;
}
