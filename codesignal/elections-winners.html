solution: 

int electionsWinners(std::vector<int> votes, int k) {
    using namespace std;
    int largest = *max_element(votes.begin(), votes.end());
    int res = 0, largest_count = 0;
    for (auto vote : votes) {
        if (vote == largest) {
            largest_count++;
        } else if (vote + k > largest) {
            res++;
        }
    }
    if (k > 0) {
        res += largest_count;
    } else if (largest_count == 1) {
        res += 1;
    }
    return res;
}
