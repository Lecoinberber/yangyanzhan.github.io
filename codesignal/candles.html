solution: 

int candles(int candlesNumber, int makeNew) {
    int res = 0, remains = 0;
    while (candlesNumber > 0) {
        res += candlesNumber;
        remains += candlesNumber;
        candlesNumber = remains / makeNew;
        remains = remains % makeNew;
    }
    return res;
}
