solution: 

int additionWithoutCarrying(int param1, int param2) {
    int res = 0, base = 1;
    while (param1 > 0 || param2 > 0) {
        int d1 = param1 % 10, d2 = param2 % 10;
        res += (d1 + d2) % 10 * base;
        base *= 10;
        param1 /= 10;
        param2 /= 10;
    }
    return res;
}
