solution: 

int neighbouringElements(std::vector<std::vector> nums) {
    using namespace std;
    int res = 0, m = nums.size(), n = nums[0].size();
    for (int i = 0; i < m - 1; i++) {
        for (int j = 0; j < n; j++) {
            if (nums[i][j] == nums[i + 1][j]) {
                res++;
            }
        }
    }
    for (int j = 0; j < n - 1; j++) {
        for (int i = 0; i < m; i++) {
            if (nums[i][j] == nums[i][j + 1]) {
                res++;
            }
        }
    }
    return res;
}
