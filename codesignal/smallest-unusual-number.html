solution: 

int smallestUnusualNumber(std::string a) {
    using namespace std;
    if (a == "0") {
        return 10;
    }
    long long sum = 0, prod = 1;
    int n_0 = 0, n_1 = 0;
    for (auto ch : a) {
        int num = ch - '0';
        if (num == 0) {
            n_0++;
        }
        if (num == 1) {
            n_1++;
        }
    }
    if (n_0 == 0) {
        for (auto ch : a) {
            int num = ch - '0';
            if (num == 1) {
                n_1--;
            }
            sum += num;
            prod *= num;
            if (prod > sum + n_1) {
                break;
            }
        }
    } else {
        sum = 1;
        prod = 0;
    }
    if (sum > prod) {
        return 0;
    }
    return 10 - (a[a.size() - 1] - '0');
}
