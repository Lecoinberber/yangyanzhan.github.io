solution: 

std::vector<std::string> unusualLexOrder(std::vector words) {
    using namespace std;
    sort(words.begin(), words.end(), [](auto item1, auto item2) {
        string s1 = item1, s2 = item2;
        reverse(s1.begin(), s1.end());
        reverse(s2.begin(), s2.end());
        return s1 < s2;
    });
    return words;
}
