bool efficientRoadNetwork(int n, std::vector<std::vector> roads) {
    using namespace std;
    map<int, set> graph;
    for (auto road : roads) {
        int a = road[0], b = road[1];
        if (graph.find(a) == graph.end()) {
            graph[a] = set<int>();
        }
        if (graph.find(b) == graph.end()) {
            graph[b] = set<int>();
        }
        graph[a].insert(b);
        graph[b].insert(a);
    }
    for (int i = 0; i < n; i++) {
        set<int> visited{i};
        if (graph.find(i) != graph.end()) {
            for (int i1 : graph[i]) {
                visited.insert(i1);
                if (graph.find(i1) != graph.end()) {
                    for (int i2 : graph[i1]) {
                        visited.insert(i2);
                    }
                }
            }
        }
        if (visited.size() < n) {
            return false;
        }
    }
    return true;
}
