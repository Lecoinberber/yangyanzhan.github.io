solution: 

#include <regex>
bool variableName(std::string name) {
    using namespace std;
    regex reg("^[\\da-zA-Z_]+$");
    smatch matches;
    regex_search(name, matches, reg);
    if (matches.size() == 0) {
        return false;
    }
    return name[0] < '0' || name[0] > '9';
}
