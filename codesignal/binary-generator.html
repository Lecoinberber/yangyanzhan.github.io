std::vector<std::string> binaryGenerator(std::string s) {
    using namespace std;
    vector<string> res;
    for (auto ch : s) {
        if (ch == '1') {
            if (res.size() > 0) {
                for (auto &item : res) {
                    item += ch;
                }
            } else {
                res = {"1"};
            }
        } else {
            if (res.size() > 0) {
                for (auto &item : res) {
                    item += '0';
                }
                int n = res.size();
                for (int i = 0; i < n; i++) {
                    string item = res[i];
                    item[item.size() - 1] = '1';
                    res.push_back(item);
                }
            } else {
                res = {"0", "1"};
            }
        }
    }
    sort(res.begin(), res.end());
    return res;
}
