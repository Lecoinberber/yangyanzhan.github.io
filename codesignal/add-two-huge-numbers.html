  • Singly-linked lists are already defined with this interface:
  • template
  • struct ListNode {
  • ListNode(const T &v) : value(v), next(nullptr) {}
  • T value;
  • ListNode *next;
  • };

solution: 

ListNode<int> *reverse(ListNode *node) {
    ListNode<int> *head = nullptr;
    while (node) {
        ListNode<int> *next = node->next;
        node->next = head;
        head = node;
        node = next;
    }
    return head;
}
ListNode<int> *addTwoHugeNumbers(ListNode *a, ListNode *b) {
    ListNode<int> *nums1 = reverse(a), *nums2 = reverse(b), *sum = nullptr;
    int carry = 0, base = 10000;
    while (nums1 || nums2) {
        int added = 0;
        if (nums1) {
            added += nums1->value;
            nums1 = nums1->next;
        }
        if (nums2) {
            added += nums2->value;
            nums2 = nums2->next;
        }
        added += carry;
        carry = added / base;
        added = added % base;
        ListNode<int> *node = new ListNode(added);
        node->next = sum;
        sum = node;
    }
    if (carry) {
        ListNode<int> *node = new ListNode(carry);
        node->next = sum;
        sum = node;
    }
    return sum;
}
