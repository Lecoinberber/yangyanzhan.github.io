bool isHarshad(long long num) {
    using namespace std;
    int sum = 0;
    string item = to_string(num);
    for (int i = 0; i < item.size(); i++) {
        sum += item[i] - '0';
    }
    return num % sum == 0;
}
long long moduloHarshad(long long num1, long long num2) {
    int c = 0, a = 0, b = 0, num = 0;
    while (c < max(num1, num2)) {
        num++;
        if (isHarshad(num)) {
            c++;
            if (c == num1) {
                a = num;
            }
            if (c == num2) {
                b = num;
            }
        }
    }
    return b % a;
}
