bool sudoku(std::vector<std::vector> grid) {
    using namespace std;
    for (int i = 0; i < 9; i++) {
        set<int> cache;
        for (int j = 0; j < 9; j++) {
            cache.insert(grid[i][j]);
        }
        if (cache.size() != 9) {
            return false;
        }
    }
    for (int j = 0; j < 9; j++) {
        set<int> cache;
        for (int i = 0; i < 9; i++) {
            cache.insert(grid[i][j]);
        }
        if (cache.size() != 9) {
            return false;
        }
    }
    for (int i1 = 0; i1 < 3; i1++) {
        for (int j1 = 0; j1 < 3; j1++) {
            set<int> cache;
            for (int i2 = 0; i2 < 3; i2++) {
                for (int j2 = 0; j2 < 3; j2++) {
                    int i = i1 * 3 + i2;
                    int j = j1 * 3 + j2;
                    cache.insert(grid[i][j]);
                }
            }
            if (cache.size() != 9) {
                return false;
            }
        }
    }
    return true;
}
