solution: 

bool almostIncreasingSequence(std::vector<int> sequence) {
    int n = sequence.size();
    int pre = sequence[0];
    bool has_seen_non_ascend = false;
    for (int i = 1; i < n; i++) {
        int now = sequence[i];
        if (pre >= now) {
            if (has_seen_non_ascend) {
                return false;
            }
            has_seen_non_ascend = true;
            if (i >= 2) {
                int ppre = sequence[i - 2];
                if (ppre < min(pre, now)) {
                    pre = min(pre, now);
                }
            } else {
                pre = min(pre, now);
            }
        } else {
            pre = now;
        }
    }
    return true;
}
