solution: 

std::vector<std::string>
allLongestStrings(std::vector<std::string> inputArray) {
    std::vector<std::string> res;
    for (auto item : inputArray) {
        if (res.empty()) {
            res.push_back(item);
        } else {
            if (item.size() == res[0].size()) {
                res.push_back(item);
            } else if (item.size() > res[0].size()) {
                res.clear();
                res.push_back(item);
            }
        }
    }
    return res;
}
