solution: 

std::vector<double> quadraticEquation(int a, int b, int c) {
    using namespace std;
    double delta = (double)b * b - 4.0 * a * c;
    if (delta < 0) {
        return {};
    }
    if (delta == 0) {
        return {-b / (2.0 * a)};
    }
    delta = sqrt(delta);
    vector<double> res{(-b - delta) / (2.0 * a), (-b + delta) / (2.0 * a)};
    sort(res.begin(), res.end());
    return res;
}
