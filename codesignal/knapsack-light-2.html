solution: 

std::string knapsackLight2(int weight1, int weight2, int maxW) {
    using namespace std;
    if (weight1 + weight2 <= maxW) {
        return "both";
    }
    if (weight1 <= maxW && weight2 <= maxW) {
        return "either";
    }
    if (weight1 <= maxW) {
        return "first";
    }
    if (weight2 <= maxW) {
        return "second";
    }
    return "none";
}
