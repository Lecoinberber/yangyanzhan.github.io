solution: 

char fractionComparison(std::vector<int> a, std::vector b) {
    using namespace std;
    double a1 = a[0], a2 = a[1];
    double b1 = b[0], b2 = b[1];
    int base = 10000;
    long long r1 = a1 / a2 * base;
    long long r2 = b1 / b2 * base;
    if (r1 == r2) {
        return '=';
    } else if (r1 < r2) {
        return '<';
    } else {
        return '>';
    }
}
