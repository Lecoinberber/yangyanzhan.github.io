std::string rome(int n) {
    using namespace std;
    vector<int> bases = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
    vector<string> names = {"M",  "CM", "D",  "CD", "C",  "XC", "L",
                            "XL", "X",  "IX", "V",  "IV", "I"};
    string res;
    int i = 0;
    while (n) {
        int base = bases[i];
        string name = names[i];
        while (n >= base) {
            n -= base;
            res += name;
        }
        i++;
    }
    return res;
}
