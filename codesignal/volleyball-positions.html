solution: 

int Q(int x) {
    int f = x % 6;
    return (f < 0 ? f + 6 : f);
}
std::vector<std::vector>
volleyballPositions(std::vector<std::vector> f, int k) {
    std::vector<std::string> p = {f[3][2], f[1][2], f[0][1],
                                  f[1][0], f[3][0], f[2][1]};
    f[3][2] = p[Q(-k)];
    f[1][2] = p[Q(-k + 1)];
    f[0][1] = p[Q(-k + 2)];
    f[1][0] = p[Q(-k + 3)];
    f[3][0] = p[Q(-k + 4)];
    f[2][1] = p[Q(-k + 5)];
    return f;
}
