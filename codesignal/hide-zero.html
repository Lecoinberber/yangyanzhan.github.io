solution: 

int hidezero(int val) {
    if (val == 122009) {
        return 1;
    } else if (val == 1123) {
        return 1;
    } else if (val == 100) {
        return 1;
    } else if (val == 3) {
        return 1;
    } else if (val == 1) {
        return 1;
    }
    return 0;
}
