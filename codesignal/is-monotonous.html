bool isMonotonous(std::vector<int> sequence) {
    using namespace std;
    if (sequence.size() <= 1) {
        return true;
    }
    if (sequence.size() == 2) {
        return sequence[0] != sequence[1];
    }
    for (int i = 2; i < sequence.size(); i++) {
        if (sequence[i - 2] < sequence[i - 1] &&
            sequence[i - 1] < sequence[i]) {
        } else if (sequence[i - 2] > sequence[i - 1] &&
                   sequence[i - 1] > sequence[i]) {
        } else {
            return false;
        }
    }
    return true;
}
