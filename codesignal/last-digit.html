solution: 

int lastDigit(int a, int b) {
    using namespace std;
    a %= 10;
    int num = 1;
    vector<int> nums;
    for (int i = 1; i <= b; i++) {
        num *= a;
        num %= 10;
        auto it = find(nums.begin(), nums.end(), num);
        if (it == nums.end()) {
            nums.push_back(num);
        } else {
            int idx = distance(nums.begin(), it);
            int head = idx, len = nums.size() - idx;
            int offset = (b - head - 1) % len;
            num = nums[idx + offset];
            break;
        }
    }
    return num;
}
