solution: 

bool checkIncreasingSequence(std::vector<int> seq) {
    if (seq.size() <= 1) {
        return true;
    }
    for (int i = 1; i < seq.size(); i++) {
        if (seq[i] <= seq[i - 1]) {
            return false;
        }
    }
    return true;
}
