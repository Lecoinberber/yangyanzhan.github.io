bool beautifulText(std::string inputString, int l, int r) {
    for (int n = l; n <= r; n++) {
        if (inputString.size() % (n + 1) == 0) {
        } else if ((inputString.size() + 1) % (n + 1) == 0) {
        } else {
            continue;
        }
        bool valid = true;
        for (int i = 0; i < inputString.size() / (n + 1); i++) {
            if (inputString[(i + 1) * (n + 1) - 1] != ' ') {
                valid = false;
                break;
            }
        }
        if (valid) {
            return true;
        }
    }
    return false;
}
