solution: 

bool namingRoads(std::vector<std::vector> roads) {
    using namespace std;
    map<int, set> cache;
    for (auto road : roads) {
        int a = road[0], b = road[1], n = road[2];
        if (cache.find(n - 1) != cache.end()) {
            if (cache[n - 1].find(a) != cache[n - 1].end()) {
                return false;
            }
            if (cache[n - 1].find(b) != cache[n - 1].end()) {
                return false;
            }
        }
        if (cache.find(n + 1) != cache.end()) {
            if (cache[n + 1].find(a) != cache[n + 1].end()) {
                return false;
            }
            if (cache[n + 1].find(b) != cache[n + 1].end()) {
                return false;
            }
        }
        cache[n] = set<int>{a, b};
    }
    return true;
}
