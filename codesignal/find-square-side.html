solution: 

int findSquareSide(std::vector<int> x, std::vector y) {
    using namespace std;
    int res = 0, count = 0;
    for (int i = 0; i < x.size(); i++) {
        for (int j = i + 1; j < x.size(); j++) {
            int dx = x[i] - x[j], dy = y[i] - y[j];
            double dist = sqrt(dx * dx + dy * dy);
            int size = dist * dist;
            if (count == 0) {
                res = size;
                count++;
            } else {
                if (res == size) {
                    count++;
                } else {
                    count--;
                }
            }
        }
    }
    return res;
}
