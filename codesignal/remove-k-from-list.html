  • Singly-linked lists are already defined with this interface:
  • template
  • struct ListNode {
  • ListNode(const T &v) : value(v), next(nullptr) {}
  • T value;
  • ListNode *next;
  • };

solution: 

ListNode<int> *removeKFromList(ListNode *l, int k) {
    using namespace std;
    ListNode<int> *head = new ListNode();
    head->next = l;
    ListNode<int> *pre = head, *now = l;
    while (now != nullptr) {
        ListNode<int> *next = now->next;
        if (now->value == k) {
            delete now;
            now = next;
            pre->next = now;
        } else {
            pre = now;
            now = now->next;
        }
    }
    return head->next;
}
