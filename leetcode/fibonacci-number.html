solution: 

class Solution {
    vector<int> cache = vector(31, 0);
  public:
    int fib(int N) {
        if (N == 0) {
            return 0;
        }
        if (N == 1) {
            return 1;
        }
        if (cache[N] == 0) {
            cache[N] = fib(N - 1) + fib(N - 2);
        }
        return cache[N];
    }
};
