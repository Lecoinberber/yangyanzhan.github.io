solution: 

class Solution {
  public:
    vector<vector> groupThePeople(vector &groupSizes) {
        map<int, vector>> groups;
        for (int i = 0; i < groupSizes.size(); i++) {
            int size = groupSizes[i];
            if (groups.find(size) == groups.end()) {
                groups[size] = vector<vector>();
            }
            bool found = false;
            for (auto &group : groups[size]) {
                if (group.size() < size) {
                    group.push_back(i);
                    found = true;
                    break;
                }
            }
            if (!found) {
                groups[size].push_back({i});
            }
        }
        vector<vector> res;
        for (auto pair : groups) {
            for (auto group : pair.second) {
                res.push_back(group);
            }
        }
        return res;
    }
};
