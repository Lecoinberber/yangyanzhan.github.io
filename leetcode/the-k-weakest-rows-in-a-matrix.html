solution: 

class Solution {
  public:
    vector<int> kWeakestRows(vector> &mat, int k) {
        int m = mat.size(), n = mat[0].size();
        vector<vector> tuples;
        for (int i = 0; i < m; i++) {
            int c = 0;
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 0) {
                    break;
                }
                c++;
            }
            tuples.push_back({c, i});
        }
        sort(tuples.begin(), tuples.end());
        vector<int> res;
        for (int i = 0; i < k; i++) {
            res.push_back(tuples[i][1]);
        }
        return res;
    }
};
