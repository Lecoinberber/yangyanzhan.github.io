solution: 

class Solution {
    set<int> visited = {0};
    vector<int> path = {0};
  public:
    vector<vector> allPathsSourceTarget(vector> &graph) {
        vector<vector> res;
        int n = graph.size(), last = path[path.size() - 1];
        if (last == n - 1) {
            res.push_back(path);
            return res;
        }
        for (auto next : graph[last]) {
            if (visited.find(next) == visited.end()) {
                visited.insert(next);
                path.push_back(next);
                auto sub_res = allPathsSourceTarget(graph);
                res.insert(res.end(), sub_res.begin(), sub_res.end());
                visited.erase(next);
                path.pop_back();
            }
        }
        return res;
    }
};
