solution: 

class Solution {
  public:
    int countBattleships(vector<vector> &board) {
        int res = 0;
        int m = board.size(), n = board[0].size();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] != 'X') {
                    continue;
                } else {
                    res++;
                    board[i][j] = 'O';
                    bool is_row = false;
                    for (int j1 = j + 1; j1 < n; j1++) {
                        if (board[i][j1] != 'X') {
                            break;
                        } else {
                            board[i][j1] = 'O';
                            is_row = true;
                        }
                    }
                    if (!is_row) {
                        for (int i1 = i + 1; i1 < m; i1++) {
                            if (board[i1][j] != 'X') {
                                break;
                            } else {
                                board[i1][j] = 'O';
                            }
                        }
                    }
                }
            }
        }
        return res;
    }
};
