solution: 

using namespace std;
class Solution {
  public:
    string convert(string s, int numRows) {
        if (numRows == 1) {
            return s;
        }
        vector<string> items(numRows);
        for (int i = 0; i < s.size(); i++) {
            int pos = i % (2 * numRows - 2);
            if (pos >= numRows) {
                pos = numRows - 1 - (pos - numRows + 1);
            }
            items[pos] = items[pos] + s[i];
        }
        string res;
        for (auto item : items) {
            res += item;
        }
        return res;
    }
};
