solution: 

class Solution {
  public:
    vector<int> replaceElements(vector &arr) {
        if (arr.size() == 0) {
            return arr;
        }
        if (arr.size() == 1) {
            arr[0] = -1;
            return arr;
        }
        int peak = arr[arr.size() - 1];
        arr[arr.size() - 1] = -1;
        for (int i = arr.size() - 2; i >= 0; i--) {
            int tmp = max(peak, arr[i]);
            arr[i] = peak;
            peak = tmp;
        }
        return arr;
    }
};
