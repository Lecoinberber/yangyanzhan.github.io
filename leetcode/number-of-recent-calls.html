solution: 

class RecentCounter {
    int pre = 0;
    vector<int> times;
  public:
    RecentCounter() {}
    int ping(int t) {
        times.push_back(t);
        int n = times.size();
        for (int i = pre; i < n; i++) {
            if ((t - 3000) <= times[i]) {
                pre = i;
                return n - i;
            }
        }
        return 0;
    }
};
/**
 * Your RecentCounter object will be instantiated and called as such:
 * RecentCounter* obj = new RecentCounter();
 * int param_1 = obj->ping(t);
 */
