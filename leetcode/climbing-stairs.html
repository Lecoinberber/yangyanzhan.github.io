solution: 

class Solution {
    map<int, int> cache;
  public:
    int climbStairs(int n) {
        if (n < 0) {
            return 0;
        }
        if (n <= 1) {
            return 1;
        }
        if (cache.find(n) != cache.end()) {
            return cache[n];
        }
        int res = climbStairs(n - 1) + climbStairs(n - 2);
        cache[n] = res;
        return res;
    }
};
