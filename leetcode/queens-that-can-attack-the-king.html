solution: 

class Solution {
  public:
    vector<vector> queensAttacktheKing(vector> &queens,
                                            vector<int> &king) {
        vector<vector> board(8, vector(8, 0));
        vector<vector> res;
        for (auto &queen : queens) {
            board[queen[0]][queen[1]] = 1;
        }
        int i = king[0], j = king[1];
        for (int di = -1; di <= 1; di++) {
            for (int dj = -1; dj <= 1; dj++) {
                if (di == 0 && dj == 0) {
                    continue;
                }
                for (int k = 1; k < 8; k++) {
                    int i1 = i + k * di, j1 = j + k * dj;
                    if (0 <= i1 && i1 < 8 && 0 <= j1 && j1 < 8) {
                        if (board[i1][j1] == 1) {
                            res.push_back({i1, j1});
                            break;
                        }
                    } else {
                        break;
                    }
                }
            }
        }
        return res;
    }
};
