solution: 

class Solution {
    vector<int> to_binary(int num) {
        vector<int> res;
        while (num > 0) {
            int digit = num % 2;
            res.push_back(digit);
            num /= 2;
        }
        return res;
    }
  public:
    int hammingDistance(int x, int y) {
        auto digits1 = to_binary(x);
        auto digits2 = to_binary(y);
        int m = digits1.size(), n = digits2.size();
        int res = 0, i;
        for (i = 0; i < min(m, n); i++) {
            if (digits1[i] != digits2[i]) {
                res++;
            }
        }
        for (; i < m; i++) {
            if (digits1[i] != 0) {
                res++;
            }
        }
        for (; i < n; i++) {
            if (digits2[i] != 0) {
                res++;
            }
        }
        return res;
    }
};
