solution: 

class Solution {
  public:
    vector<vector> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector> res;
        int i = r0, j = c0;
        int i_dir = 0, i_step = 0, step = 1;
        vector<vector> dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        while (true) {
            if (res.size() == R * C) {
                break;
            }
            if (0 <= i && i < R && 0 <= j && j < C) {
                res.push_back({i, j});
            }
            if (i_step == step) {
                if (i_dir == 1 || i_dir == 3) {
                    step += 1;
                }
                i_step = 0;
                i_dir++;
                i_dir %= 4;
            }
            i_step++;
            i += dirs[i_dir][0];
            j += dirs[i_dir][1];
        }
        return res;
    }
};
