solution: 

class Solution {
  public:
    vector<vector> fourSum(vector &nums, int target) {
        set<vector> tuples;
        sort(nums.begin(), nums.end());
        int n = nums.size();
        for (int i = 0; i <= n - 4; i++) {
            int a = nums[i];
            for (int j = i + 1; j <= n - 3; j++) {
                int b = nums[j];
                int k = j + 1, l = n - 1;
                while (k < l) {
                    int c = nums[k], d = nums[l];
                    int sum = a + b + c + d;
                    if (sum == target) {
                        k++;
                        l--;
                        tuples.insert({a, b, c, d});
                    } else if (sum < target) {
                        k++;
                    } else {
                        l--;
                    }
                }
            }
        }
        vector<vector> res(tuples.begin(), tuples.end());
        return res;
    }
};
