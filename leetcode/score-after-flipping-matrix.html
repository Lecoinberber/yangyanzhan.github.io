solution: 

class Solution {
  public:
    int matrixScore(vector<vector> &A) {
        int m = A.size(), n = A[0].size();
        for (int i = 0; i < m; i++) {
            if (A[i][0] == 0) {
                for (int j = 0; j < n; j++) {
                    A[i][j] = 1 - A[i][j];
                }
            }
        }
        int res = 0;
        int base = 1;
        for (int j = n - 1; j > 0; j--) {
            int count = 0;
            for (int i = 0; i < m; i++) {
                if (A[i][j] == 1) {
                    count++;
                }
            }
            count = max(count, m - count);
            res += count * base;
            base *= 2;
        }
        res += m * base;
        return res;
    }
};
