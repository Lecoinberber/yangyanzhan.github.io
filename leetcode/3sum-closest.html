solution: 

class Solution {
  public:
    int threeSumClosest(vector<int> &nums, int target) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        int res = 0, min_gap = numeric_limits<int>::max();
        for (int i = 0; i <= n - 3; i++) {
            int a = nums[i];
            int j = i + 1, k = n - 1;
            while (j < k) {
                int b = nums[j], c = nums[k];
                int sum = a + b + c;
                int gap = sum - target;
                int abs_gap = abs(gap);
                if (abs_gap < min_gap) {
                    res = sum;
                    min_gap = abs_gap;
                }
                if (gap == 0) {
                    return res;
                } else if (gap > 0) {
                    k--;
                } else {
                    j++;
                }
            }
        }
        return res;
    }
};
