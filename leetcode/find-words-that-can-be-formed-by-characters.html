solution: 

class Solution {
  public:
    int countCharacters(vector<string> &words, string chars) {
        map<char, int> cache;
        for (auto ch : chars) {
            if (cache.find(ch) == cache.end()) {
                cache[ch] = 0;
            }
            cache[ch]++;
        }
        int res = 0;
        for (auto &word : words) {
            map<char, int> tmp = cache;
            bool valid = true;
            for (auto ch : word) {
                if (tmp.find(ch) == tmp.end() || tmp[ch] == 0) {
                    valid = false;
                    break;
                } else {
                    tmp[ch]--;
                }
            }
            if (valid) {
                res += word.size();
            }
        }
        return res;
    }
};
