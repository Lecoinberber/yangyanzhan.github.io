solution: 

class Solution {
  public:
    int oddCells(int n, int m, vector<vector> &indices) {
        vector<vector> cache(n);
        for (int i = 0; i < n; i++) {
            cache[i] = vector<int>(m, 0);
        }
        for (auto pair : indices) {
            int di = pair[0], dj = pair[1];
            for (int j = 0; j < m; j++) {
                cache[di][j]++;
            }
            for (int i = 0; i < n; i++) {
                cache[i][dj]++;
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (cache[i][j] % 2 != 0) {
                    res++;
                }
            }
        }
        return res;
    }
};
