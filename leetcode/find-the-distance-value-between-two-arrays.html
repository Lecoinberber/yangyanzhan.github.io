solution: 

class Solution {
  public:
    int findTheDistanceValue(vector<int> &arr1, vector &arr2, int d) {
        int res = 0;
        for (auto num1 : arr1) {
            bool valid = true;
            for (auto num2 : arr2) {
                if (abs(num1 - num2) <= d) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                res++;
            }
        }
        return res;
    }
};
