solution: 

class Solution {
  public:
    int balancedStringSplit(string s) {
        int res = 0, lc = 0, rc = 0;
        while (s.size() > 0) {
            if (s[0] == 'L') {
                lc++;
            } else {
                rc++;
            }
            for (int i = 1; i < s.size(); i++) {
                if (s[i] == 'L') {
                    lc++;
                } else {
                    rc++;
                }
                if (lc == rc) {
                    break;
                }
            }
            res++;
            s = s.substr(lc + rc);
            lc = 0;
            rc = 0;
        }
        return res;
    }
};
