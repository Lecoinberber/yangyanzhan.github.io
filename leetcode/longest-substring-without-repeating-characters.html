solution: 

class Solution {
  public:
    int lengthOfLongestSubstring(string s) {
        int start = 0, max_len = 0;
        unordered_map<char, int> cache;
        for (int i = 0; i < s.size(); i++) {
            auto ch = s[i];
            if (cache.find(ch) == cache.end()) {
                cache[ch] = i;
            } else {
                int pre = cache[ch];
                if (pre >= start) {
                    max_len = max(max_len, i - start);
                    start = pre + 1;
                }
                cache[ch] = i;
            }
        }
        max_len = max(max_len, (int)s.size() - start);
        return max_len;
    }
};
