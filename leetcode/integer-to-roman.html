solution: 

class Solution {
  public:
    string intToRoman(int num) {
        vector<string> names = {"M",  "CM", "D",  "CD", "C",  "XC", "L",
                                "XL", "X",  "IX", "V",  "IV", "I"};
        vector<int> values = {1000, 900, 500, 400, 100, 90, 50,
                              40,   10,  9,   5,   4,   1};
        string res;
        for (int i = 0; i < values.size(); i++) {
            int value = values[i];
            string name = names[i];
            for (int j = 0; j < num / value; j++) {
                res += name;
            }
            num -= num / value * value;
        }
        return res;
    }
};
