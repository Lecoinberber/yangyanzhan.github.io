solution: 

using namespace std;
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
  public:
    ListNode *removeNthFromEnd(ListNode *head, int n) {
        ListNode *res = head;
        ListNode *p0 = nullptr, *p1 = head, *p2 = head;
        for (int i = 0; i < n; i++) {
            p2 = p2->next;
        }
        while (p2 != nullptr) {
            p0 = p1;
            p1 = p1->next;
            p2 = p2->next;
        }
        if (p0 == nullptr) {
            return res->next;
        }
        p0->next = p1->next;
        return res;
    }
};
