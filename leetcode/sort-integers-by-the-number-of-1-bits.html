solution: 

class Solution {
    map<int, int> cache;
    int value(int n) {
        if (cache.find(n) == cache.end()) {
            int n1 = n, res = 0;
            while (n1 > 0) {
                res += n1 % 2;
                n1 /= 2;
            }
            cache[n] = res;
        }
        return cache[n];
    }
  public:
    vector<int> sortByBits(vector &arr) {
        sort(arr.begin(), arr.end(), [this](int n1, int n2) {
            int v1 = value(n1), v2 = value(n2);
            if (v1 != v2) {
                return v1 < v2;
            }
            return n1 < n2;
        });
        return arr;
    }
};
