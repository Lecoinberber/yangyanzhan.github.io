solution: 

int fact(int n) {
    long long res = 1;
    for (int i = 1; i <= n; i++) {
        res *= i;
        res %= 1000003;
    }
    return res;
}
int Solution::findRank(string A) {
    int n = A.length();
    int res = 0;
    for (int i = 0; i < n - 1; i++) {
        int smaller = 0;
        for (int j = i + 1; j < n; j++) {
            if (A[j] < A[i])
                smaller++;
        }
        res += smaller * fact(n - i - 1);
        res %= 1000003;
    }
    return res + 1;
}
