solution: 

vector<int> Solution::prevSmaller(vector &A) {
    int n = A.size();
    vector<int> res(n), indexes(n, -1);
    if (n == 0)
        return res;
    indexes[0] = -1;
    for (int i = 1; i < n; i++) {
        int j = i - 1;
        while (j >= 0) {
            if (A[j] < A[i]) {
                indexes[i] = j;
                break;
            } else {
                int index = indexes[j];
                if (index == -1) {
                    indexes[i] = -1;
                    break;
                } else {
                    j = index;
                }
            }
        }
    }
    for (int i = 0; i < n; i++) {
        if (indexes[i] == -1) {
            res[i] = -1;
        } else {
            res[i] = A[indexes[i]];
        }
    }
    return res;
}
