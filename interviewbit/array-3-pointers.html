solution: 

int Solution::minimize(const vector<int> &A, const vector &B,
                       const vector<int> &C) {
    int i = 0, j = 0, k = 0, n1 = A.size(), n2 = B.size(), n3 = C.size(),
        best = INT_MAX;
    while (i < n1 && j < n2 && k < n3) {
        int a = A[i], b = B[j], c = C[k];
        int bottom = min(a, min(b, c));
        int top = max(a, max(b, c));
        int diff = top - bottom;
        if (diff == 0) {
            return 0;
        }
        best = min(best, diff);
        if (a == bottom) {
            i++;
        } else if (b == bottom) {
            j++;
        } else {
            k++;
        }
    }
    return best;
}
