solution: 

int Solution::colorful(int A) {
    stringstream ss;
    ss << A;
    string word = ss.str();
    set<int> nums;
    for (int i = 0; i < word.length(); i++) {
        for (int k = 1; i + k <= word.length(); k++) {
            string part = word.substr(i, k);
            long long product = 1;
            for (int j = 0; j < part.length(); j++) {
                product *= (part[j] - '0');
            }
            if (nums.find(product) != nums.end()) {
                return 0;
            }
            nums.insert(product);
        }
    }
    return 1;
}
