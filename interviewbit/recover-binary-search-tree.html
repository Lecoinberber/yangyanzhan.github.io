solution: 

TreeNode *pre;
TreeNode *first;
TreeNode *second;
void inorder(TreeNode *root) {
    if (!root) {
        return;
    }
    inorder(root->left);
    if (!pre) {
        pre = root;
    } else {
        int n1 = pre->val, n2 = root->val;
        if (n1 > n2) {
            if (!first) {
                first = pre;
            }
            second = root;
        }
        pre = root;
    }
    inorder(root->right);
}
vector<int> Solution::recoverTree(TreeNode *A) {
    pre = NULL, first = NULL, second = NULL;
    vector<int> res;
    inorder(A);
    if (first && second) {
        res.push_back(second->val);
        res.push_back(first->val);
    }
    return res;
}
