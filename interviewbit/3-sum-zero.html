solution: 

string getKey(int i, int j, int k, vector<int> &A) {
    stringstream ss;
    ss << A[i] << " " << A[j] << " " << A[k];
    return ss.str();
}
vector<vector> Solution::threeSum(vector &A) {
    set<string> seen;
    sort(A.begin(), A.end());
    int n = A.size();
    vector<vector> res;
    for (int i = 0; i < n - 2; i++) {
        int begin = i + 1, end = n - 1;
        while (begin < end) {
            long long sum = A[i];
            sum += A[begin];
            sum += A[end];
            if (sum == 0) {
                string key = getKey(i, begin, end, A);
                if (seen.find(key) == seen.end()) {
                    seen.insert(key);
                    res.push_back(vector<int>{A[i], A[begin], A[end]});
                }
                begin++;
                end--;
            } else if (sum < 0) {
                begin++;
            } else {
                end--;
            }
        }
    }
    return res;
}
