solution: 

vector<int> Solution::slidingMaximum(const vector &A, int B) {
    vector<int> res;
    vector<int> nums;
    int n = A.size();
    for (int i = 0; i < n; i++) {
        int a = A[i];
        if (nums.empty()) {
            nums.push_back(a);
        } else {
            int tail = nums.back();
            while (tail < a) {
                nums.pop_back();
                if (nums.empty()) {
                    break;
                }
                tail = nums.back();
            }
            nums.push_back(a);
            if (i + 1 > B) {
                if (A[i - B] == nums.front()) {
                    nums.erase(nums.begin());
                }
            }
        }
        if (i + 1 >= B) {
            res.push_back(nums.front());
        }
    }
    return res;
}
