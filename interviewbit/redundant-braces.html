solution: 

int Solution::braces(string A) {
    stack<char> tokens;
    string ops = "+-*/";
    for (char ch : A) {
        if (ch != ')') {
            tokens.push(ch);
        } else {
            int cnt = 0;
            while (!tokens.empty() && tokens.top() != '(') {
                char token = tokens.top();
                if (ops.find(token) != string::npos) {
                    cnt++;
                }
                tokens.pop();
            }
            if (tokens.empty()) {
                return 0;
            }
            tokens.pop();
            if (cnt <= 0) {
                return 1;
            }
        }
    }
    return 0;
}
