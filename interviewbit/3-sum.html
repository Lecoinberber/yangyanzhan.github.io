int Solution::threeSumClosest(vector<int> &A, int B) {
    sort(A.begin(), A.end());
    int n = A.size(), best = 0, bestDiff = INT_MAX;
    for (int i = 0; i < n - 2; i++) {
        int begin = i + 1, end = n - 1;
        long long sum;
        while (begin < end) {
            sum = A[i];
            sum += A[begin];
            sum += A[end];
            if (sum == B) {
                return B;
            } else {
                if (abs(sum - B) < bestDiff) {
                    best = sum;
                    bestDiff = (int)abs(sum - B);
                }
            }
            if (sum < B) {
                begin++;
            } else {
                end--;
            }
        }
    }
    return best;
}
