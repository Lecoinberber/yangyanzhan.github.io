solution: 

int Solution::climbStairs(int A) {
    int a = 1, b = 2;
    if (A <= 2) {
        return A;
    }
    for (int i = 3; i <= A; i++) {
        int tmp = b;
        b = a + b;
        a = tmp;
    }
    return b;
}
