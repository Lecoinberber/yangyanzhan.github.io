solution: 

vector<vector> Solution::diagonal(vector> &A) {
    int n = A.size();
    vector<vector> res;
    for (int k = 0; k < n; k++) {
        vector<int> row;
        for (int i = 0; i <= k; i++) {
            row.push_back(A[i][k - i]);
        }
        res.push_back(row);
    }
    for (int k = 1; k < n; k++) {
        vector<int> row;
        for (int i = k; i < n; i++) {
            row.push_back(A[i][n + k - 1 - i]);
        }
        res.push_back(row);
    }
    return res;
}
