solution: 

TreeNode *build(vector<int> &preorder, int pBegin, int pEnd,
                vector<int> &inorder, int iBegin, int iEnd) {
    if (pBegin > pEnd) {
        return NULL;
    }
    int num = preorder[pBegin];
    int idx = find(inorder.begin() + iBegin, inorder.begin() + iEnd + 1, num) -
              inorder.begin();
    TreeNode *root = new TreeNode(num);
    int len1 = idx - iBegin, len2 = iEnd - idx;
    root->left = build(preorder, pBegin + 1, pBegin + len1, inorder, iBegin,
                       iBegin + len1 - 1);
    root->right = build(preorder, pBegin + len1 + 1, pBegin + len1 + len2,
                        inorder, idx + 1, idx + len2);
    return root;
}
TreeNode *Solution::buildTree(vector<int> &preorder, vector &inorder) {
    return build(preorder, 0, preorder.size() - 1, inorder, 0,
                 inorder.size() - 1);
}
