map<int, int> getFactors(int n) {
    int i = 2;
    map<int, int> res;
    while (n > 1) {
        if (n % i == 0) {
            int count = 0;
            while (n % i == 0) {
                count++;
                n /= i;
            }
            res[i] = count;
        }
        i++;
    }
    return res;
}
int gcd(int a, int b) {
    if (a % b == 0)
        return b;
    return gcd(b, a % b);
}
int Solution::isPower(int n) {
    if (n <= 1)
        return true;
    map<int, int> factors = getFactors(n);
    int g = 0;
    for (auto it : factors) {
        int factor = it.first, count = it.second;
        if (g == 0) {
            g = count;
        } else {
            g = gcd(max(g, count), min(g, count));
        }
    }
    return g > 1;
}
