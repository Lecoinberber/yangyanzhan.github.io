solution: 

void Solution::merge(vector<int> &A, vector &B) {
    int m = B.size(), i = 0;
    for (int j = 0; j < m; j++) {
        int b = B[j];
        while (i < A.size()) {
            if (A[i] > b) {
                break;
            }
            i++;
        }
        if (i < A.size()) {
            A.insert(A.begin() + i, b);
        } else {
            A.push_back(b);
        }
    }
}
