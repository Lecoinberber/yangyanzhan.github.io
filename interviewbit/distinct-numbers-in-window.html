solution: 

vector<int> Solution::dNums(vector &A, int B) {
    int n = A.size();
    vector<int> res;
    if (n < B) {
        return res;
    }
    map<int, int> cache;
    for (int i = 0; i < B; i++) {
        int num = A[i];
        if (cache.find(num) == cache.end()) {
            cache[num] = 0;
        }
        cache[num]++;
    }
    res.push_back(cache.size());
    for (int i = 1; i <= n - B; i++) {
        int pre = A[i - 1], now = A[B + i - 1];
        cache[pre]--;
        if (cache.find(now) == cache.end()) {
            cache[now] = 0;
        }
        cache[now]++;
        if (cache[pre] == 0) {
            cache.erase(pre);
        }
        res.push_back(cache.size());
    }
    return res;
}
