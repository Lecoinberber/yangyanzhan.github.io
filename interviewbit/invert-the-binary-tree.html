solution: 

TreeNode *Solution::invertTree(TreeNode *root) {
    if (!root) {
        return NULL;
    }
    TreeNode *left = root->left;
    TreeNode *right = root->right;
    root->left = right;
    root->right = left;
    invertTree(root->left);
    invertTree(root->right);
    return root;
}
