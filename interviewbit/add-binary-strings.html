string Solution::addBinary(string A, string B) {
    stringstream ss;
    int i = A.length() - 1, j = B.length() - 1, carry = 0;
    while (i >= 0 && j >= 0) {
        int digit = A[i] - '0' + B[j] - '0' + carry;
        ss << (digit % 2);
        carry = digit / 2;
        i--;
        j--;
    }
    while (i >= 0) {
        int digit = A[i] - '0' + carry;
        ss << (digit % 2);
        carry = digit / 2;
        i--;
    }
    while (j >= 0) {
        int digit = B[j] - '0' + carry;
        ss << (digit % 2);
        carry = digit / 2;
        j--;
    }
    if (carry) {
        ss << carry;
    }
    string res = ss.str();
    reverse(res.begin(), res.end());
    return res;
}
