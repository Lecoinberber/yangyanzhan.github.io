bool feasible(const vector<int> &nums, int students, int top) {
    int page = 0, i = 0, n = nums.size();
    while (i < n) {
        if (students <= 0) {
            return false;
        }
        if (page + nums[i] > top) {
            page = 0;
            students--;
        } else {
            page += nums[i];
            i++;
        }
    }
    return true;
}
int Solution::books(vector<int> &A, int B) {
    int sum = 0, n = A.size();
    if (n < B)
        return -1;
    for (int i = 0; i < n; i++)
        sum += A[i];
    int begin = sum / n, end = sum, best = -1;
    while (begin <= end) {
        int middle = (begin + end) / 2;
        if (feasible(A, B, middle)) {
            best = middle;
            end = middle - 1;
        } else {
            begin = middle + 1;
        }
    }
    return best;
}
