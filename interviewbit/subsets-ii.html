set<string> keys;
string getKey(vector<int> &nums) {
    stringstream ss;
    for (int num : nums) {
        ss << num << "-";
    }
    return ss.str();
}
void traverse(vector<int> used, vector remains, vector> &res) {
    if (remains.size() <= 0)
        return;
    int remain = remains[0];
    remains.erase(remains.begin());
    used.push_back(remain);
    string key = getKey(used);
    if (keys.find(key) == keys.end()) {
        res.push_back(used);
        keys.insert(key);
    }
    traverse(used, remains, res);
    used.pop_back();
    traverse(used, remains, res);
}
vector<vector> Solution::subsetsWithDup(vector &A) {
    keys.clear();
    sort(A.begin(), A.end());
    vector<int> used, remains = A;
    vector<vector> res = {vector()};
    traverse(used, remains, res);
    return res;
}
