void traverse(vector<int> used, vector remains, int k,
              vector<vector> &res) {
    if (used.size() == k) {
        res.push_back(used);
        return;
    }
    if (remains.size() <= 0) {
        return;
    }
    int remain = remains[0];
    remains.erase(remains.begin());
    used.push_back(remain);
    traverse(used, remains, k, res);
    used.pop_back();
    traverse(used, remains, k, res);
}
vector<vector> Solution::combine(int n, int k) {
    vector<vector> res;
    if (n < k || k <= 0)
        return res;
    vector<int> used, remains;
    for (int i = 1; i <= n; i++) {
        remains.push_back(i);
    }
    traverse(used, remains, k, res);
    return res;
}
