solution: 

int Solution::uniquePaths(int A, int B) {
    int n = A - 1 + B - 1, i = max(A - 1, B - 1) + 1, j = min(A - 1, B - 1),
        m = 1;
    long long res = 1;
    for (int k = i; k <= n; k++) {
        res *= k;
        if (m <= j && res % m == 0) {
            res /= m;
            m++;
        }
    }
    for (int k = m; k <= j; k++) {
        res /= k;
    }
    return res;
}
