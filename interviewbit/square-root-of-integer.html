solution: 

int Solution::sqrt(int A) {
    double gap = 0.01, begin = 0, end = A;
    int res;
    while (true) {
        double middle = (begin + end) / 2;
        double middle2 = middle * middle;
        if (abs(middle2 - A) <= gap) {
            res = (int)floor(middle);
            break;
        }
        if (middle2 < A)
            begin = middle;
        else
            end = middle;
    }
    if ((res + 1) * (res + 1) == A)
        res++;
    return res;
}
