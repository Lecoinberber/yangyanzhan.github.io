solution: 

vector<int> Solution::repeatedNumber(const vector &A) {
    int n = A.size();
    long long sum = 0, sum2 = 0;
    long long total = n, total2 = n;
    total = total * (n + 1) / 2;
    total2 = total2 * (n + 1) * (2 * n + 1) / 6;
    for (int i = 0; i < n; i++) {
        int num = A[i];
        sum = sum + num;
        sum2 = sum2 + (long long)(num)*num;
    }
    long long b_a = total - sum;
    long long b_a2 = total2 - sum2;
    long long b_add_a = b_a2 / b_a;
    int a = (b_add_a - b_a) / 2;
    int b = (b_a + b_add_a) / 2;
    vector<int> res;
    res.push_back(a);
    res.push_back(b);
    return res;
}
