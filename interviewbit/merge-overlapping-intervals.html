solution: 

/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
bool cmp(Interval i1, Interval i2) { return i1.start < i2.start; }
vector<Interval> Solution::merge(vector &A) {
    vector<Interval> res;
    sort(A.begin(), A.end(), cmp);
    vector<Interval>::iterator it = A.begin();
    Interval i = *it;
    it++;
    while (it != A.end()) {
        Interval i1 = *it;
        if (i1.start > i.end) {
            res.push_back(i);
            i = i1;
        } else {
            i.end = max(i.end, i1.end);
        }
        it++;
    }
    res.push_back(i);
    return res;
}
