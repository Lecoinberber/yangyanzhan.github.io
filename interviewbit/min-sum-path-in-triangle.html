solution: 

int Solution::minimumTotal(vector<vector> &triangle) {
    int m = triangle.size();
    if (m <= 0) {
        return 0;
    }
    int n = triangle[m - 1].size();
    if (n <= 0) {
        return 0;
    }
    vector<int> cache = triangle[m - 1];
    for (int i = 1; i <= m - 1; i++) {
        for (int j = 0; j < n - i; j++) {
            cache[j] = min(cache[j], cache[j + 1]) + triangle[m - 1 - i][j];
        }
    }
    return cache[0];
}
