solution: 

vector<vector> Solution::solve(int A) {
    vector<vector> res;
    vector<int> pre = {1};
    vector<int> now = {1, 1};
    if (A >= 1) {
        res.push_back(pre);
    }
    if (A >= 2) {
        res.push_back(now);
    }
    if (A >= 3) {
        for (int i = 3; i <= A; i++) {
            pre = now;
            now.clear();
            now.push_back(1);
            for (int j = 1; j < pre.size(); j++) {
                now.push_back(pre[j - 1] + pre[j]);
            }
            now.push_back(1);
            res.push_back(now);
        }
    }
    return res;
}
