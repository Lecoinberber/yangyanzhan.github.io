int Solution::atoi(const string A) {
    long res = 0, sign = 1;
    bool foundDigit = false, foundSign = false;
    for (int i = 0; i < A.length(); i++) {
        char ch = A[i];
        if (!('0' <= ch && ch <= '9')) {
            if (ch == ' ') {
                if (foundDigit) {
                    break;
                }
            } else if (ch == '-' || ch == '+') {
                if (foundSign) {
                    break;
                } else {
                    if (ch == '-') {
                        sign = -1;
                    }
                    foundDigit = true;
                    foundSign = true;
                }
            } else {
                break;
            }
        } else {
            foundDigit = true;
            foundSign = true;
            res = res * 10 + (ch - '0');
            if (sign * res > INT_MAX) {
                return INT_MAX;
            } else if (sign * res < INT_MIN) {
                return INT_MIN;
            }
        }
    }
    return sign * res;
}
