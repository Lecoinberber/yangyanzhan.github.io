solution: 

int Solution::numDistinct(string s, string t) {
    int m = s.length(), n = t.length();
    vector<vector> cache(m + 1, vector(n + 1, 0));
    cache[0][0] = 1;
    for (int i = 1; i <= m; i++) {
        cache[i][0] = 1;
        for (int j = 1; j <= n; j++) {
            cache[i][j] = cache[i - 1][j];
            if (s[i - 1] == t[j - 1]) {
                cache[i][j] += cache[i - 1][j - 1];
            }
        }
    }
    return cache[m][n];
}
