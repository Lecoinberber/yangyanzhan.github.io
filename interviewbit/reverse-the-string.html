solution: 

string Solution::solve(string A) {
    vector<string> words;
    int n = A.length();
    string pre = "";
    for (int i = 0; i < n; i++) {
        char ch = A[i];
        if (ch == ' ') {
            if (pre.length() > 0) {
                words.push_back(pre);
            }
            pre = "";
        } else {
            pre += ch;
        }
    }
    if (pre.length() > 0) {
        words.push_back(pre);
    }
    A = "";
    for (int i = words.size() - 1; i >= 0; i--) {
        A += words[i];
        if (i > 0) {
            A += " ";
        }
    }
    return A;
}
