solution: 

int find1(const vector<int> &A, int B) {
    int n = A.size(), begin = 0, end = n - 1, found = -1;
    while (begin <= end) {
        int middle = (begin + end) / 2;
        if (A[middle] == B) {
            found = middle;
            end = middle - 1;
        } else if (A[middle] > B) {
            end = middle - 1;
        } else {
            begin = middle + 1;
        }
    }
    return found;
}
int find2(const vector<int> &A, int B) {
    int n = A.size(), begin = 0, end = n - 1, found = -1;
    while (begin <= end) {
        int middle = (begin + end) / 2;
        if (A[middle] == B) {
            found = middle;
            begin = middle + 1;
        } else if (A[middle] > B) {
            end = middle - 1;
        } else {
            begin = middle + 1;
        }
    }
    return found;
}
vector<int> Solution::searchRange(const vector &A, int B) {
    return vector<int>{find1(A, B), find2(A, B)};
}
