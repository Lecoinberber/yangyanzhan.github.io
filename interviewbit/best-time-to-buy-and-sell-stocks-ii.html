solution: 

int Solution::maxProfit(const vector<int> &A) {
    int n = A.size(), best = 0;
    for (int i = 0; i < n - 1; i++) {
        int profit = A[i + 1] - A[i];
        if (profit > 0) {
            best += profit;
        }
    }
    return best;
}
