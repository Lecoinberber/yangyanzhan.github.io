solution: 

void Solution::sortColors(vector<int> &A) {
    int n = A.size(), begin = 0, mid = 0, end = n - 1;
    while (mid < end + 1) {
        if (A[mid] == 0) {
            swap(A[begin], A[mid]);
            begin++, mid++;
        } else if (A[mid] == 1) {
            mid++;
        } else {
            swap(A[mid], A[end]);
            end--;
        }
    }
}
