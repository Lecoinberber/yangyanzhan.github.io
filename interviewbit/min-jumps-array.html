solution: 

int Solution::jump(vector<int> &A) {
    int n = A.size(), begin = 0, end = 0, steps = 0;
    while (begin <= end) {
        if (end >= n - 1) {
            return steps;
        }
        int newBegin = end + 1, newEnd = end;
        for (int i = begin; i <= end; i++) {
            newEnd = max(newEnd, i + A[i]);
        }
        begin = newBegin;
        end = newEnd;
        steps++;
    }
    return -1;
}
