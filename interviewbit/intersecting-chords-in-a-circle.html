solution: 

int Solution::chordCnt(int A) {
    vector<long long> cache(A + 1, 1);
    for (int i = 2; i <= A; i++) {
        cache[i] = 0;
        for (int j = 0; j <= i - 1; j++) {
            cache[i] += cache[j] * cache[i - 1 - j];
            cache[i] %= 1000000007;
        }
    }
    return cache[A];
}
