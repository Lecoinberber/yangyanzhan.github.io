int divide(string &A) {
    int remain = 0;
    stringstream ss;
    for (int i = 0; i < A.length(); i++) {
        remain = remain * 10 + (A[i] - '0');
        int digit = 0;
        if (remain >= 2) {
            digit = remain / 2;
            remain = remain % 2;
        }
        if (digit == 0) {
            if (ss.str().length() > 0) {
                ss << digit;
            }
        } else {
            ss << digit;
        }
    }
    A = ss.str();
    return remain;
}
int Solution::power(string A) {
    if (A == "1")
        return 0;
    while (A != "0" && A != "1") {
        int remain = divide(A);
        if (remain != 0) {
            return 0;
        }
    }
    return A == "1";
}
