vector<vector> Solution::subsets(vector &A) {
    sort(A.begin(), A.end());
    vector<vector> res{vector()};
    if (A.size() <= 0)
        return res;
    int head = A[0];
    A.erase(A.begin());
    vector<vector> nextRes = subsets(A);
    for (vector<int> item : nextRes) {
        item.insert(item.begin(), head);
        res.push_back(item);
    }
    for (vector<int> item : nextRes) {
        if (item.size() > 0) {
            res.push_back(item);
        }
    }
    return res;
}
