string getIp(vector<string> nums) {
    string res;
    for (string num : nums) {
        if (num[0] == '0' && num.length() > 1) {
            return "";
        }
        if (num == "") {
            return "";
        }
        int n = stoi(num);
        if (n > 255) {
            return "";
        }
        res += num + ".";
    }
    return res.substr(0, res.length() - 1);
}
void check(vector<string> &seen, string A, vector &ips) {
    if (seen.size() == 3) {
        seen.push_back(A);
        string ip = getIp(seen);
        if (ip != "") {
            ips.push_back(ip);
        }
        seen.pop_back();
    } else {
        int top = A.length();
        if (top > 3)
            top = 3;
        for (int i = 1; i <= top; i++) {
            seen.push_back(A.substr(0, i));
            check(seen, A.substr(i), ips);
            seen.pop_back();
        }
    }
}
vector<string> Solution::restoreIpAddresses(string A) {
    vector<string> res;
    vector<string> seen;
    check(seen, A, res);
    sort(res.begin(), res.end());
    return res;
}
