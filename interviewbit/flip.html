vector<int> Solution::flip(string A) {
    int n = A.length();
    int preI = 0, preJ = 0, pre = A[0] == '0' ? 1 : -1;
    int bestI = preI, bestJ = preJ, best = pre;
    for (int i = 1; i < n; i++) {
        int num = A[i] == '0' ? 1 : -1;
        if (pre >= 0) {
            pre += num;
            preJ = i;
        } else {
            pre = num;
            preI = i;
            preJ = i;
        }
        if (pre > best) {
            best = pre;
            bestI = preI;
            bestJ = preJ;
        }
    }
    vector<int> res;
    if (best > 0) {
        res.push_back(bestI + 1);
        res.push_back(bestJ + 1);
    }
    return res;
}
