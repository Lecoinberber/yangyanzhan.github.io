void traverse(vector<int> used, vector remains, int sum, int target,
              vector<vector> &res) {
    if (remains.size() <= 0)
        return;
    int remain = remains[0];
    remains.erase(remains.begin());
    while (sum < target) {
        traverse(used, remains, sum, target, res);
        used.push_back(remain);
        sum += remain;
    }
    if (sum == target) {
        res.push_back(used);
    }
}
vector<vector> Solution::combinationSum(vector &A, int B) {
    sort(A.begin(), A.end());
    vector<int>::iterator it = unique(A.begin(), A.end());
    A.resize(it - A.begin());
    vector<int> used, remains = A;
    vector<vector> res;
    int sum = 0, target = B;
    traverse(used, remains, sum, target, res);
    reverse(res.begin(), res.end());
    return res;
}
