string Solution::countAndSay(int n) {
    if (n == 1)
        return "1";
    string word = countAndSay(n - 1);
    int pre = word[0] - '0', count = 1, now;
    stringstream ss;
    for (int i = 1; i <= word.length(); i++) {
        bool diff = false;
        if (i == word.length()) {
            diff = true;
        } else {
            now = word[i] - '0';
            if (now != pre) {
                diff = true;
            }
        }
        if (diff) {
            ss << count << pre;
            count = 1;
        } else {
            count++;
        }
        pre = now;
    }
    return ss.str();
}
