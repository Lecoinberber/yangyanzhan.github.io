solution: 

void Solution::setZeroes(vector<vector> &A) {
    int m = A.size(), n = A[0].size();
    set<int> zero_rows;
    set<int> zero_cols;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (A[i][j] == 0) {
                if (zero_cols.find(j) == zero_cols.end()) {
                    for (int i1 = 0; i1 < m; i1++) {
                        if (A[i1][j] == 1) {
                            A[i1][j] = -1;
                        }
                    }
                }
                if (zero_rows.find(i) == zero_rows.end()) {
                    for (int j1 = 0; j1 < n; j1++) {
                        if (A[i][j1] == 1) {
                            A[i][j1] = -1;
                        }
                    }
                }
                zero_cols.insert(j);
                zero_rows.insert(i);
            }
        }
    }
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (A[i][j] == -1) {
                A[i][j] = 0;
            }
        }
    }
}
