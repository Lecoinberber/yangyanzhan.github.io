solution: 

int Solution::cntBits(vector<int> &A) {
    long long base = 1000000007;
    long long res = 0;
    for (int i = 0; i < 32; i++) {
        int f = 1 << i;
        long long n1 = 0, n2 = 0;
        for (int num : A) {
            if ((num & f) != 0) {
                n1++;
            } else {
                n2++;
            }
        }
        res += ((n1 % base) * ((n2) % base) * 2) % base;
    }
    return res % base;
}
