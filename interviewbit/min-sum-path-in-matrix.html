solution: 

int Solution::minPathSum(vector<vector> &grid) {
    int m = grid.size();
    if (m == 0) {
        return 0;
    }
    int n = grid[0].size();
    if (n == 0) {
        return 0;
    }
    vector<vector> cache = grid;
    for (int j = 1; j < n; j++) {
        cache[0][j] += cache[0][j - 1];
    }
    for (int i = 1; i < m; i++) {
        cache[i][0] += cache[i - 1][0];
        for (int j = 1; j < n; j++) {
            cache[i][j] += min(cache[i - 1][j], cache[i][j - 1]);
        }
    }
    return cache[m - 1][n - 1];
}
