solution: 

vector<vector> Solution::pathSum(TreeNode *root, int sum) {
    if (!root) {
        return vector<vector>();
    }
    int num = root->val;
    if (!root->left && !root->right) {
        if (num == sum) {
            return vector<vector>{vector{num}};
        }
    }
    vector<vector> leftPaths = pathSum(root->left, sum - num);
    vector<vector> rightPaths = pathSum(root->right, sum - num);
    vector<vector> res;
    for (vector<int> path : leftPaths) {
        path.insert(path.begin(), num);
        res.push_back(path);
    }
    for (vector<int> path : rightPaths) {
        path.insert(path.begin(), num);
        res.push_back(path);
    }
    return res;
}
