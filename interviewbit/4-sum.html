solution: 

string toKey(int a, int b, int c, int d) {
    stringstream ss;
    ss << a << "~~";
    ss << b << "~~";
    ss << c << "~~";
    ss << d << "~~";
    return ss.str();
}
vector<vector> Solution::fourSum(vector &A, int B) {
    vector<vector> res;
    set<string> keys;
    sort(A.begin(), A.end());
    int n = A.size();
    for (int i = 0; i < n - 3; i++) {
        for (int j = i + 1; j < n - 2; j++) {
            int a = A[i], b = A[j], begin = j + 1, end = n - 1;
            while (begin < end) {
                int c = A[begin], d = A[end];
                int sum = a + b + c + d;
                if (sum == B) {
                    string key = toKey(a, b, c, d);
                    if (keys.find(key) == keys.end()) {
                        keys.insert(key);
                        res.push_back(vector<int>{a, b, c, d});
                    }
                    begin++;
                    end--;
                } else if (sum < B) {
                    begin++;
                } else {
                    end--;
                }
            }
        }
    }
    return res;
}
