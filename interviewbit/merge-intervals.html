solution: 

/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
vector<Interval> Solution::insert(vector &intervals,
                                  Interval newInterval) {
    vector<Interval> res;
    vector<Interval>::iterator it = intervals.begin();
    while (it != intervals.end()) {
        Interval interval = *it;
        int x1 = newInterval.start, y1 = newInterval.end, x2 = interval.start,
            y2 = interval.end;
        if (y1 < x2) {
            res.push_back(newInterval);
            break;
        }
        if ((x1 <= x2 && x2 <= y1) || (x2 <= x1 && x1 <= y2)) {
            newInterval.start = min(x1, x2);
            newInterval.end = max(y1, y2);
        } else {
            res.push_back(interval);
        }
        it++;
    }
    if (it == intervals.end()) {
        res.push_back(newInterval);
    }
    while (it != intervals.end()) {
        res.push_back(*it);
        it++;
    }
    return res;
}
