unit class Bob;
method hey ($msg) {
    given $msg {
        when /^<[A..Z\s]>+\?$/ { "Calm down, I know what I'm doing!" }
        when /^\s*$/ { "Fine. Be that way!" }
        when /\?\s*$/ { "Sure." }
        when /^<-[a..z]>*<[A..Z]><-[a..z]>*$/ { "Whoa, chill out!" }
        default { "Whatever." }
    }
}
