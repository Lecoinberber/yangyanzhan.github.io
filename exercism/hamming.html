unit module Hamming;
sub hamming-distance ($strand1, $strand2) is export {
    die "left and right strands must be of equal length" if $strand1.chars != $strand2.chars;
    return (0..^$strand1.chars).grep({ $strand1.comb[$_] !eq $strand2.comb[$_] }).elems;
}
